	tag = "characters_regular"

	###SAILORS
	illustration_image = {
		texture_file = "gfx/interface/illustrations/event/characters/nobles/north_german/north_german_nobles_regular.dds"
		priority = 6
		scope = country
		trigger = {
			event_illustration_foreground_trigger = yes
			scope:target_culture ?= { gfx_culture_applicable = european_gfx } # better for other cultures to use peasants than european sailors
			scope:target_pop_type ?= {
				OR = {
					this = pop_type:laborers
					this = pop_type:peasants
					AND = {
						this = pop_type:soldiers
						modifier:local_sailors >= modifier:local_manpower
					}
				}
				location = { 
					is_port = yes

				}
			}
