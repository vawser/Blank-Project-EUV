# Material indices need to fit into a single channel of a compressed texture
# That means a maximum of 32 material-textures are supported. If you have multiple materials in this file that use the same texture they will count as one
#
#
#	example = {
#		texture = "flatmap_hills.dds"	-- which texture to use for this materials
#		priority = 50					-- Only one material will be used per pixel. If two materials are valid for a pixel the one with higher priority is used
#
#		topography = "mountains"			-- Which types of topography this material is valid for. If no topography is entered then this material is valid for all topographies
#		topography = "mountain_wasteland" -- If more than one topography is entered then this material is valid for all those topographies.
#
#		vegetation = "jungle"			-- Same as with the topography. 
#		vegetation = " flatland_wasteland" -- If both topography and vegetation are entered then the material is valid if the location has at least one topography match and at least one vegetation match
#
#		climate = "tropical"
#		climate = "continental"
#	}
