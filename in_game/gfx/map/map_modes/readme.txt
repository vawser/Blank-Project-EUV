# MAP MODES
#
# color_mode = <mode> color mode (if not using a custom mode)
# map_color = <script color> map color for location (root = location, scope:actor/recipient etc)
# secondary_map_color = <script color> striped map color for location (root = location, scope:actor/recipient etc)
# tooltip_key = <script string key> scripted key for the tooltip localization (root = location, scope:actor/recipient etc)
# small_map_names = what to text to show on the map
# medium_map_names = what to text to show on the map
# large_map_names = what to text to show on the map
# small_tooltip_context = tooltip context for small map
# medium_tooltip_context = tooltip context for medium map
# large_tooltip_context = tooltip context for large map
# color_and_names_refresh_counters = list of counters that will cause a refresh of the map colors and names
# color_refresh_counters = list of counters that will cause a refresh of the map colors only
# refresh_colors_on_selection_change = yes/no whether or not colors will be refreshed when selection changes
# fill_in_impassable = yes/no whether or not to fill impassable terrain in
# enable_snow = yes/no show snow or not
# enable_water_names = yes/no show names of seas and oceans
# enable_rivers = yes/no whether or not to show rivers
# use_highlight = yes/no whether or not to show highlights
# use_fow = yes/no whether or not to show fog of war
# allow_allocate_hotkey = yes/no whether or not to allow allocating a hot key to this map mode
# quick_action_slot = <integer> slot used for this map mode in quick actions
# map_lines_mode = what type of map lines to show (hard coded)
# override_border_rendering = overrides of border rendering
# map_markers = map markers to allow or disallow
# flatmap_behaviour = how to show flat map
# flatmap_gradient_parameters = custom handling of flat map colors
