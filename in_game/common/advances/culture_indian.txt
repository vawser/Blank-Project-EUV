paik_system_advance = {
	age = age_1_traditions
	potential = {
		OR = {
			has_or_had_tag = AHO
			has_or_had_tag = MLB
			AND = {
				culture = { has_culture_group = culture_group:indian_group }
				capital ?= {
					OR = {
						area = area:bengal_area
						area = area:assam_area
					}
				}
			}
		}
	}
	unlock_levy = levy_a_early_paik
	requires = unlock_feudal_levy_advance
}

samanta_advance = {
	age = age_1_traditions
	potential = { culture = { has_culture_group = culture_group:indian_group } }
	unlock_subject_type = samanta
	unlock_subject_type = maha_samanta
	unlock_subject_type = pradhana_maha_samanta
	requires = feudalism_advance
