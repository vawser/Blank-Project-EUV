hre_improve_with_electors = {
    days_between_checks = 360
    actor_trigger = { 
        unique_international_organization_type_exists = international_organization_type:hre
        can_lead_international_organization = international_organization:hre
        OR = {

            #Elector
            international_organization:hre = {
                OR = {
                    country_has_special_status = {
                        type = special_status:elector
                        country = root
                    }
                    country_has_special_status = {
                        type = special_status:archbishop_elector
                        country = root
                    }
                }
            }

            #Prince in HRE
            AND = {          
                is_member_of_international_organization = international_organization:hre
                country_rank_level >= 2
            }

            #Great power close to HRE
            AND = {
                is_great_power = yes
                international_organization:hre = {
                    leader_country ?= {
                        "border_distance_to(root)" < 300
                    }
                }
            }

            #Current emperor
            is_leader_of_international_organization = international_organization:hre

            #Somebody is voting for us     
            international_organization:hre = {
                "active_resolution(resolution:hre_election)" = {
                    save_temporary_scope_as = election
                }
                OR = {
                    any_country_with_special_status_of_type = {
                        type = elector
                        scope:election = {
                            "cast_vote_in_active_resolution(prev)" ?= root 
                        }
                    }
                    any_country_with_special_status_of_type = {
                        type = archbishop_elector
                        scope:election = {
                            "cast_vote_in_active_resolution(prev)" ?= root 
                        }
                    }
                }
            }
        }
    }
    recipient_trigger = {
        NOT = {
            scope:recipient = {
                is_rival_of = scope:actor
            }
            scope:actor = {
                is_rival_of = scope:recipient
            }
            international_organization:hre = {
                "active_resolution(resolution:hre_election)" = {
                    "cast_vote_in_active_resolution(scope:recipient)" ?= scope:actor 
                }
            }
        }
    }
    recipient_priority = {
        add = "scope:recipient.opinion(scope:actor)"
        add = 50
    }
    recipient_list_builder = { 
        international_organization:hre = {
            country_with_special_status_of_type = {
                type = elector
            }
        }
        international_organization:hre = {
            country_with_special_status_of_type = {
                type = archbishop_elector
            }
        }
    }
    pause_trigger = { 
        scope:actor = {
            at_war = yes
        }
    }
    cancel_trigger = { 
        international_organization:hre = {
            "active_resolution(resolution:hre_election)" = {
                "cast_vote_in_active_resolution(scope:recipient)" ?= scope:actor 
            }
        }
    }
    country_relations = {
        alliance = yes
    }
    country_interactions = {
        ask_to_vote_for_me_as_emperor = yes
    }
    improve_relation = yes
    max_allowed = 2

}
