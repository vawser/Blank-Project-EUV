wako_sponsorship =  {
	type = diplomacy
	relation_type = oneway	
	break_on_war = yes
	fleet_basing_rights = yes
	receiving_color = rgb { 60 120 0 }
	giving_color = rgb { 40 100 0 }
	category = {
		offer = CATEGORY_FRIENDLY_ACTIONS
		request = CATEGORY_FRIENDLY_ACTIONS
		cancellation = CATEGORY_HOSTILE_ACTIONS
		break = CATEGORY_HOSTILE_ACTIONS
	}

	relation_type_for_ai = friendly

	visible = {
		scope:actor = {
			is_subject = no
			NOT = { is_wokou_nation = yes }
			culture = { has_culture_group = culture_group:japanese_group }
		}
		scope:recipient = {
			is_subject = no
			culture = { has_culture_group = culture_group:japanese_group }
			is_wokou_nation = yes
		}
	}

	offer_enabled = {
		
	}

	offer_effect = {
		scope:recipient = {
			set_variable = {
				name = sponsoring_tag
				value = scope:actor
			}
		}
	}
	cancel_effect = {
		scope:recipient = {
			remove_variable = sponsoring_tag
		}
	}
	break_effect = {
		scope:recipient = {
			remove_variable = sponsoring_tag
		}
	}
	
	will_expire_trigger = {
		scope:recipient = {
			OR = {
				is_subject = yes
				NOT = { 
					culture = { has_culture_group = culture_group:japanese_group }
				}
				NOT = { is_wokou_nation = yes }
			}
		}
	}
	wants_to_give_diplo_chance = {
		actor_is_rival = -100
		base = -40
		#not_10_trust = -100
		trust_in_actor = 1
		positive_opinion = 0.5
		negative_opinion = -1
		border_distance = -0.01
		same_court_language = 10
		strategic_interest = 1
		common_threat = 3
		relative_strength = 40
	}
	wants_to_give = {
		add = {
			desc = "THEIR_OPINION_TOOLTIP"
			value = "scope:recipient.opinion(scope:actor)"
			divide = 2
		}
		add = {
			desc = "DIPLOREASON_DIPLOMATIC_REPUTATION"
			scope:actor = {
				value = modifier:diplomatic_reputation
				multiply = 5
			}
		}
		add = {
			desc = "DISTANCE_TO_TARGET"
			scope:recipient = {
				value = "border_distance_to(scope:actor)"
				multiply = -0.02
			}
		}
		add = {
			desc = "game_concept_rival"
			if = {
				limit = {	
					scope:recipient = { 
						is_rival_of = scope:actor 
					}
				}
				value = -1000
			}
		}
	}
	wants_to_receive = {
		add = {
			desc = "DIPLOREASON_GOOD_FOR_US" # Sponsorship is sought after by a wako nation
			value = 20
		}
		add = {
			desc = "THEIR_OPINION_TOOLTIP"
			value = "scope:recipient.opinion(scope:actor)"
			divide = 2
		}
		add = {
			desc = "DIPLOREASON_DIPLOMATIC_REPUTATION"
			scope:actor = {
				value = modifier:diplomatic_reputation
				multiply = 5
			}
		}
		add = {
			desc = "DISTANCE_TO_TARGET"
			scope:recipient = {
				value = "border_distance_to(scope:actor)"
				multiply = -0.02
			}
		}
		add = {
			desc = "game_concept_rival"
			if = {
				limit = {	
					scope:recipient = { 
						is_rival_of = scope:actor 
					}
				}
				value = -1000
			}
		}
	}
}
