fondaco_rights =  {
	type = diplomacy
	relation_type = oneway	
	break_on_war = yes
	category = {
		offer = CATEGORY_ECONOMY_ACTIONS
		request = CATEGORY_ECONOMY_ACTIONS
		cancellation = CATEGORY_ECONOMY_ACTIONS
		break = CATEGORY_ECONOMY_ACTIONS
	}

	relation_type_for_ai = friendly

	visible = {
		scope:recipient = {
			is_merchant_republic = yes
			culture = { has_culture_group = culture_group:italian_group }
		}
	}

	break_enabled = {
		OR = {
			scope:recipient = {
				NOT = { has_reform = government_reform:merchant_republic }
			}
			scope:actor = {
				OR = {
					giving_scripted_relation = {
						type = relation_type:deny_market_access
						target = scope:recipient
					}
					is_embargoed_by = scope:recipient
					is_embargoing = scope:recipient
					giving_scripted_relation = {
						type = relation_type:block_foreign_building
						target = scope:recipient
					}
				}
			}
		}
	}
	
	wants_to_receive = {
		add = {
			desc = "DIPLOREASON_GOOD_FOR_US"
			value = 100
		}
	}

	wants_to_give = {
		add = {
			desc = "THEIR_OPINION_TOOLTIP"
			value = "scope:recipient.opinion(scope:actor)"
			divide = 5
		}
		add = {
			desc = "DIPLOREASON_DIPLOMATIC_REPUTATION"
			scope:actor = {
				value = modifier:diplomatic_reputation
				multiply = 5
			}
		}
		subtract = {
			desc = "DIPLOREASON_BASE_RELUCTANCE"
			value = 25
		}
		if = {
			limit = {	
				scope:recipient = { 
					is_rival_of = scope:actor 
				}
			}
			subtract = {
				desc = "game_concept_rival"
				value = 100
			}
		}
		if = {
			limit = {	
				scope:recipient = { 
					has_reform = government_reform:merchant_republic
				}
			}
			subtract = {
				desc = "DIPLOREASON_COMPETING_MERCHANT_REPUBLIC"
				value = 100
			}
		}
	}
}
