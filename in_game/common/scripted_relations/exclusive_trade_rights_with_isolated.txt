exclusive_trade_rights_with_isolated =  { 
	type = diplomacy
	relation_type = oneway
	break_on_war = yes
	is_exempt_from_isolation = yes
	category = {
		offer = CATEGORY_ECONOMY_ACTIONS
		request = CATEGORY_ECONOMY_ACTIONS
		cancellation = CATEGORY_ECONOMY_ACTIONS
		break = CATEGORY_ECONOMY_ACTIONS
	}

	relation_type_for_ai = friendly

	visible = {
		scope:actor = {
			modifier:trade_isolation = yes
		}
	}

	wants_to_receive = {
		add = {
			desc = "game_concept_rival"
			if = {
				limit = {	
					scope:recipient = { 
						is_rival_of = scope:actor 
					}
				}
				value = -100
			}
		}
		
		add = {
			desc = "game_concept_isolated"
			value = 50
		}
	}
}
