trade_access =  { 
	type = diplomacy
	relation_type = oneway
	break_on_war = yes
	#first is the market owner, second is the country granted trade access
	lifts_trade_protection = yes
	trade_to_first = 0.2
	category = {
		offer = CATEGORY_ECONOMY_ACTIONS
		request = CATEGORY_ECONOMY_ACTIONS
		cancellation = CATEGORY_ECONOMY_ACTIONS
		break = CATEGORY_ECONOMY_ACTIONS
	}

	relation_type_for_ai = friendly

	visible = {
		always = yes
	}

	offer_enabled = {
		scope:actor = { has_markets = yes }
	}

	request_enabled = {
		scope:actor = { has_markets = no }
		scope:recipient = { has_markets = yes }
	}

	break_effect = {
		scope:actor = {
			add_opinion = {
				modifier = 	opinion_broke_trade_access
				target = scope:recipient
			}
		}
	}

	wants_to_give_diplo_chance = {
        base = 10 #really, you always want people to use your markets.
	}

	wants_to_receive_diplo_chance = {
        actor_is_rival = -100
		actor_overlord_is_rival = -100
        base = -40
        trust_in_actor = 1
        positive_opinion = 0.25
        negative_opinion = -1
        border_distance = -0.05
        strategic_interest = 1
	}

	wants_to_receive = {
		if = {
			limit = {
				scope:recipient = { has_markets = yes }
			}
			add = {
				desc = "trade_access_HAS_MARKETS"
				value = -200
			}
		}
	}
}
