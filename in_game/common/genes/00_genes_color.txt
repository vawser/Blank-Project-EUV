color_genes = {
	hair_color = {
		color = hair
		#max_blend = 0.0
		#max_recessive_drift = 0.25
	}
	skin_color = {
		sync_inheritance_with = hair_color
		color = skin
		# When inheriting a color gene, we pick clone either the dominant or recessive settings from one of our parents.
		# Color blending only happens when rendering the portraits, so the blended color will never get inherited
		#max_blend = 0.2
		blend_range = { 0.55 0.65 }
	}
	eye_color = {
		sync_inheritance_with = hair_color
		color = eye
	}

}
