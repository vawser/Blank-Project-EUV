
dominion = { # like a personal union except the subject has much less power
	subject_pays = subject_pays_vassal
	color = subject_dominion
	level = 3

	join_offensive_wars_always = {
		NOT = { scope:actor ?= { is_subject_of = scope:recipient } }
	}
	join_offensive_wars_can_call = {
		scope:actor ?= { is_subject_of = scope:recipient }
	}
	join_defensive_wars_always = {
		always = yes
	}
	has_overlords_ruler = yes
	annexation_speed = 1
	annexation_min_years_before = 100
	annexation_min_opinion = 150
	annexation_stall_opinion = 125
	subject_can_cancel = no
	overlord_can_cancel = yes
	has_limited_diplomacy = yes

	allow_declaring_wars = { 
		lordship_of_ireland_can_declare_wars_trigger = yes
	}

	war_score_cost = 0.5	
	
	strength_vs_overlord = -0.25
	diplomatic_capacity_cost_scale = 0.5

	only_overlord_court_language = yes
	only_overlord_or_kindred_culture = yes

	can_change_rank = no
	can_change_heir_selection = no
	
	minimum_opinion_for_offer = 175
	
	food_access = yes
	
	great_power_score_transfer = 0.5 # the amount of Great Power Score that is transferred from the Subject to the Overlord

	institution_spread_to_overlord = monthly_institution_spread_severe
	institution_spread_to_subject = monthly_institution_spread_severe
	
	overlord_modifier = {
		monthly_prestige = 0.01
	}
	subject_modifier = {
		legislative_efficiency = 0.1
	}

	creation_visible = {
		OR = {
			has_or_had_tag = ENG
			has_or_had_tag = GBR
		}
	}

	subject_creation_enabled = {
		government_type = government_type:monarchy
	}

	release_country_enabled = {
		government_type = government_type:monarchy
	}

	visible_through_diplomacy = {
		OR = {
			in_union_with = scope:target
			has_or_had_tag = ENG
			has_or_had_tag = GBR
		}
		religion.group = scope:target.religion.group
		government_type = government_type:monarchy
		country_rank_level >= scope:target.country_rank_level
		
		#Japanese Imperial Courts Exception
		trigger_if = {
			limit = {
				is_member_of_international_organization = international_organization:japanese_shogunate
				is_situation_active = situation:nanbokuchou
			}
			NOT = {
				scope:target = {
					has_special_status_in_international_organization = {
						type = special_status:japanese_emperor
						international_organization = international_organization:japanese_shogunate
					}
				}
			}
		}
	}

	visible_through_treaty = {
		OR = {
			in_union_with = scope:target
			has_or_had_tag = ENG
			has_or_had_tag = GBR
		}
		religion.group = scope:target.religion.group
		government_type = government_type:monarchy
		country_rank_level >= scope:target.country_rank_level
		
		#Japanese Imperial Courts Exception
		trigger_if = {
			limit = {
				is_member_of_international_organization = international_organization:japanese_shogunate
				is_situation_active = situation:nanbokuchou
			}
			NOT = {
				scope:target = {
					has_special_status_in_international_organization = {
						type = special_status:japanese_emperor
						international_organization = international_organization:japanese_shogunate
					}
				}
			}
		}
	}

	enabled_through_diplomacy = {
		religion.group = scope:target.religion.group
		trigger_if = {
			limit = {
				in_union_with = scope:target
			}
			union_length_days = {
				target = scope:target
				value > 18250
			}
			relative_strength = {
				target = scope:target
				value > 0.7
			}
		}
	}

	on_enable = {
		#if already in a union, then we need to get them out of it
		union ?= {
			remove_country_from_international_organization = root
		}
		if = {
			limit = {
				exists = scope:future_overlord.ruler
				NOT = { ruler ?= scope:future_overlord.ruler }
			}
			set_new_foreign_ruler = scope:future_overlord.ruler
		}
		
		if = {
			limit = {
				culture = {
					NOR = {
						this = scope:future_overlord.culture
						reverse_cultural_view = { target = scope:future_overlord.culture value = kindred }
					}					
				}
			}
			change_culture = scope:future_overlord.culture
		}

		if = {
			limit = {
				and = {
					exists = court_language
					court_language != scope:future_overlord.court_language
				}
			}
			set_court_language = scope:future_overlord.court_dialect
		}
	}

	on_disable = {
	}
	
	diplo_chance_accept_subject = {
		base = -70
		
		
		current_strength = 0.2
		border_distance = -0.3
		negative_opinion = -5
		positive_opinion = 0.25
		rank_difference = -5
		giving_defensive_support = 2
		receiving_defensive_support = 5
		royal_ties = 1

		num_of_limited_subjects = -5
		culture_war = 3
		diplomatic_reputation = 1
		different_religion = -20
		same_common_language = 5
		same_court_language = 10
		different_culture = -10
		different_government_type = -50
		actor_at_war = -20
		competing_power = -200
		tax_base = -0.25
		
		reject_subjugation_reasons = -1
		accept_subjugation_reasons = 1
	}

	diplo_chance_accept_overlord = {
		base = 12
		border_distance = -0.3
		negative_opinion = -5
		positive_opinion = 0.25
		same_common_language = 5
		same_court_language = 10
		different_culture = -10
		different_government_type = -25
		actor_at_war = -20
		competing_power = -200
	}

	ai_wants_to_be_overlord = {
        scope:subject = {
			add = {
                desc = "COALITION_GRADE_ANTAGONISM_TOWARDS_THEM_SUBJECT"
                value = {
                    every_country_with_coalition_grade_antagonism_against_us = {                    
                        add = {
                            value = -10
                        }
                    }
                }
            }
        }
	}
}
