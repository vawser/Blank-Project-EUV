
tributary = { 
	subject_pays = subject_pays_tributary
	color = subject_tributary
	level = 0
	visible_through_diplomacy = {
		country_rank_level >= scope:target.country_rank_level
		scope:target = { subject_type_is_not_locked = yes }
		OR = {
			government_type = government_type:steppe_horde
			scope:target = { government_type = government_type:tribe }
			scope:target = { government_type = government_type:steppe_horde }
			modifier:allow_tributary_subject = yes
		}
		trigger_if = {
			limit = {
				exists = international_organization:japanese_shogunate
				scope:target = { is_member_of_international_organization = international_organization:japanese_shogunate }

			}
			scope:target = { is_leader_of_international_organization = international_organization:japanese_shogunate }
		}
		
		#Japanese Imperial Courts Exception
		trigger_if = {
			limit = {
				is_member_of_international_organization = international_organization:japanese_shogunate
				is_situation_active = situation:nanbokuchou
			}
			NOT = {
				scope:target = {
					has_special_status_in_international_organization = {
						type = special_status:japanese_emperor
						international_organization = international_organization:japanese_shogunate
					}
				}
			}
		}
	}

	visible_through_treaty = {
		country_rank_level >= scope:target.country_rank_level

		OR = {
			government_type = government_type:steppe_horde
			scope:target = { government_type = government_type:tribe }
			scope:target = { government_type = government_type:steppe_horde }
			modifier:allow_tributary_subject = yes
		}
		trigger_if = {
			limit = {
				exists = international_organization:japanese_shogunate

				NOT = { is_member_of_international_organization = international_organization:japanese_shogunate }
			}
			scope:target = { is_leader_of_international_organization = international_organization:japanese_shogunate }
		}
		
		#Japanese Imperial Courts Exception
		trigger_if = {
			limit = {
				is_member_of_international_organization = international_organization:japanese_shogunate
				is_situation_active = situation:nanbokuchou
			}
			NOT = {
				scope:target = {
					has_special_status_in_international_organization = {
						type = special_status:japanese_emperor
						international_organization = international_organization:japanese_shogunate
					}
				}
			}
		}
	}

	creation_visible = {
		OR = {
			government_type = government_type:steppe_horde
			government_type = government_type:tribe
			modifier:allow_tributary_subject = yes
		}
	}
	
	join_defensive_wars_always = {
		custom_tooltip = {
			text = tributary_join_defensive_wars_always_tt
			scope:actor = {
				is_subject_of = scope:recipient
			}
			scope:target ?= {
				OR = {
					NOT = { is_subject_of = scope:recipient }
					AND = {
						is_subject_of = scope:recipient
						NOT = { is_subject_type = tributary }
					}
				}
			}
		}
	}
