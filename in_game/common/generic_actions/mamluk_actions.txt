promote_mamluks = {
	type = parliament

	sound = PARLIAMENT_PROMOTE_MAMLUKS
	
	potential = {
		scope:actor = { 
			modifier:can_promote_mamluks = yes
		}
	}


	allow = {
		scope:actor = {
			is_active_parliament = yes
			parliament_issue_support >= {
				value = parliament_request_support
				divide = 2
			}
		}
	}

	ai_tick = daily
	ai_tick_frequency = 20

	automation_tick = daily
	automation_tick_frequency = 20
	
	effect = {
		scope:actor = {
			custom_tooltip = {
				text = promote_mamluks_tt
				trigger_event_silently = flavor_mam.17
			}
			change_parliament_issue_support = {
				value = {
					add = modifier:parliament_request_issue_support_needed
					divide = 2
				}
				multiply = -1
			}
		}		
	}
	
	ai_will_do = {
		add = {
			if = {
				limit = {
					scope:actor = {
						parliament_issue_support > 0.75
					}
				}
				value = 10
			}
		}
	}
}
