hire_artist = {
	type = owncountry

	price = price:hire_artist

	ai_tick = monthly
	ai_tick_frequency = 3

	automation_tick = monthly
	automation_tick_frequency = 1
	player_automated_category = governmentreform


	potential = {
	}

	allow = {
		scope:actor = {
			num_artists < modifier:num_possible_artists
			modifier:has_cultural_maintenance = yes
		}
	}

	ai_will_do = {
		scope:actor = {
			#first, how many do we need?
			value = {
				#how many should we have?
				value = modifier:num_possible_artists
				#how many have we already got?
				subtract = num_artists
			}

			#not massively important
			divide = 5
		}
	}

	effect = {
		scope:actor = {
			custom_tooltip = {
				text = hire_artist_effect
				create_character = {
					trait_category = artist
					
					save_scope_as = new_character
				}
			}
		}
	}
}
