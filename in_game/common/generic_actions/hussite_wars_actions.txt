convert_heretics_hw_action = {
	type = situation
	show_message = no
	potential = {
		scope:actor = {
			religion = religion:hussite
		}
	}

	allow = {
		scope:actor = {
			any_province = { 
				NOT = { has_province_modifier = hw_heretic_conversions_modifier }
				any_location_in_province = {
					is_ownable = yes
					owner = scope:actor
					OR = {
						dominant_religion ?= { this != scope:actor.religion }
						local_religious_unity < 1.0
					}
				}
			}
		}
	}
	
	ai_tick = monthly
	ai_tick_frequency = 4
	price = price:hussite_wars_actions_price

	automation_tick = never
	automation_tick_frequency = 12

	cooldown = {
		type = convert_heretics_hw_action
		years = 1
	}

	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "choose_situation"
		column = {
			data = name
		}
		visible = {
			situation:hussite_wars = this
			situation_is_active = yes
		}
	}

	select_trigger = {
		looking_for_a = religion
		target_flag = target_religion
		source = actor
		name = "promote_religion_select_religion"
		none_available_msg_key = "convert_religion_no_religions"
		column = {
			data = name
		}
		column = {
			data = religion_percentage_in_country
		}
		default_sort = religion_percentage_in_country
		visible = {
			OR = {
				this = religion:hussite
				this = religion:catholic
			}
		}
	}

	select_trigger = {
		looking_for_a = province
		target_flag = target
		source = actor
		name = "choose_province"
		none_available_msg_key = "no_provinces_available"
		column = { data = name }
		column = { data = province_religion_percentage }
		column = { data = population }
		visible = {
			religion_percentage = {
				religion = scope:target_religion
				value < 1
			}
		}
		enabled = {
			NOT = { has_province_modifier = hw_heretic_conversions_modifier }
		}
		map_color = {
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_LOW
				max_color = define:NMapColors|MAP_COLOR_HIGH
				factor = "province.religion_percentage(scope:target_religion)"
			}
		}
	}
	effect = { 
		if = {
			limit = {
				exists = scope:target
			}
			scope:target = {
				add_province_modifier = {
					modifier = hw_heretic_conversions_modifier
					mode = add_and_extend
					years = 10
					size = {
						value = 1
						add = {
							add = "scope:actor.estate_power(estate_type:clergy_estate)"
							desc = ESTATE_OPINION_TARGET_C_ESTATE_POWER
						}
					}
				}
			}
		}
		else = {
			custom_tooltip = hw_province_gains_hw_heretic_conversions_modifier_tt
		}
		
	}

	ai_will_do = {
		add = 5

		if = {
			limit = { exists = scope:target }
			add = scope:target.population
			divide = scope:actor.total_population
			multiply = 100
		}

		add = scope:actor.religious_influence
	}
}

		}

	effect = {
