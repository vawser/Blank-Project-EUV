orthodox_synod = {
	type = religious

	sound = UI_action_religion_generic
	
	potential = {
		scope:actor.religion = religion:orthodox
	}

	allow = {
	}

	cooldown = {
		type = orthodox_synod
		years = 10
	}
	
	price = price:orthodox_synod

	ai_tick = monthly
	ai_tick_frequency = 1

	automation_tick = monthly
	automation_tick_frequency = 1
	player_automated_category = religiousdoctrines

	select_trigger = {
		looking_for_a = religion
		source = actor
		target_flag = recipient
		name = "choose_religion"
		column = {
			data = name
		}
	}

	effect = {
		scope:actor = {
			trigger_event_non_silently = orthodox_flavor.100
		}
		
	}

	ai_will_do = {
		#base
		value = 25
		#don't do it when other, more expensive options are available
		if = {
			limit = {
				scope:actor = {
					OR = {
						can_do_generic_action = {
							generic_action = generic_action:third_rome
							recipient = scope:actor.religion
						}
						can_do_generic_action = {
							generic_action = generic_action:create_autocephalous_patriarchate
						}
					}
				}
			}
			multiply = 0
		}
	}
}

periphora = {
	type = owncountry

	sound = UI_action_religion_generic
	
	potential = {
		scope:actor = {
			religion = religion:orthodox
		}
	}

	allow = {
	}

	ai_tick = monthly
	ai_tick_frequency = 1

	automation_tick = monthly
	automation_tick_frequency = 1
	player_automated_category = religiousdoctrines
	
	price = price:periphora

	#icon to move
	select_trigger = {
		looking_for_a = work_of_art
		source = actor
		target_flag = work_of_art
		name = "periphora_select_icon"
		none_available_msg_key = "periphora_no_works_of_art"
		column = {
			data = name
		}
		visible = {
			trigger_if = {
				limit = {
					exists = scope:location
				}
				not = { location = scope:location }
			}
			work_of_art_type = work_of_art_type:icon
			has_periphora = no
		}
	}

	#destination
	select_trigger = {
		looking_for_a = location
		source = actor
		target_flag = location
		cache_targets = yes #this list of targets doesn't rely on the previous target so we only need to do it once
		name = "periphora_select_destination"
		none_available_msg_key = "periphora_no_locations"
		column = {
			data = name
		}
		column = {
			data = population
		}
		visible = {
			trigger_if = {
				limit = {
					exists = scope:work_of_art
				}
				not = {
					scope:work_of_art = {
						location = root
					}
				}
			}
		}
	}

	effect = {
		if = {
			limit = {
				exists = scope:work_of_art
				exists = scope:location
			}
			scope:work_of_art = {
				start_periphora_towards = scope:location
			}
		}
	}
	
	ai_will_do = {
		#inertia
		add = -10
		
		#is the new location better suited to having art than the old one?
		add = {
			value = scope:location.modifier:local_unrest
			subtract = scope:work_of_art.location.modifier:local_unrest
			multiply = 10
		}
	}
}
