change_employment_system = {
	type = owncountry

	potential = {
	}

	allow = {
	}

	ai_tick = monthly
	ai_tick_frequency = 60

	automation_tick = monthly
	automation_tick_frequency = 1
	player_automated_category = buildings
	
	price = price:change_employment_system

	select_trigger = {
		looking_for_a = employment_system
		target_flag = target
		name = "change_employment_system_choose_system"
		column = {
			data = name
		}
		enabled = {
            scope:actor = {
                has_advance_for_employment_system = root
				NOT = {
					has_employment_system = root
				}
            }			
		}
	}

	effect = {
		scope:actor = {
			if = {
				limit = {
					exists = scope:target
				}
				set_country_employment_system = scope:target
			}
		}
	}

	ai_will_do = {
		#inertia
		add = {
			desc = "inertia"
			value = -30
		}

        if = {
            limit = {
                exists = scope:target
            }
            add = {
                value = "scope:actor.employment_system_desire(scope:target)"
            }
        }
	}
}
