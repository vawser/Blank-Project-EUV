change_military_stance = {
	type = owncountry

	potential = {
	}

	allow = {
	}

	ai_tick = never #handled in code

	select_trigger = {
		looking_for_a = country
		source = actor
		source_flags = subjects
		target_flag = subject
		name = "change_military_stance_choose_subject"
		column = {
			data = name
		}
		column = {
			data = army_size
		}
		column = {
			data = regular_army_size
		}
		column = {
			data = navy_size
		}
		visible = {
		}
	}

	select_trigger = {
		looking_for_a = military_stance
		target_flag = military_stance
		name = "change_military_stance_choose_stance"
		column = {
			data = name
		}
        selected = {
            this = scope:subject.country_stance
        }
	}

	effect = {
		scope:actor = {
			if = {
				limit = {
					exists = scope:subject
					exists = scope:military_stance
				}
                scope:subject = {
				    set_country_military_stance = scope:military_stance
                }
			}
		}
	}
}
