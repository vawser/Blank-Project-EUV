claim_ilkhanate = {
	type = internationalorganization

	ai_tick = daily
	ai_tick_frequency = 40

	automation_tick = never
	automation_tick_frequency = 12


	potential = {
		scope:recipient = { international_organization_type = international_organization_type:ilkhanate }
	}

	allow = {
		scope:actor = {
			custom_tooltip = {
				text = we_were_not_forced_to_renounce_ilkhan_claim_tt
				NOT = { has_variable = forced_to_renounce_ilkhan_claim }
			}
			NOT = {
				has_special_status_in_international_organization = {
					type = special_status:ilkhan_claimant
					international_organization = international_organization:ilkhanate
				}
			}
			OR = {
				ruler_or_heir_if_regent ?= { is_dynastic_descendant_of = dynasty:borjigin_dynasty }
				any_character = { is_dynastic_descendant_of = dynasty:borjigin_dynasty }
			}
		}
	}

	price = government_power_mild_penalty

	select_trigger = {
		looking_for_a = international_organization
		source = actor
		target_flag = recipient
		name = "choose_international_organization"
		column = {
			data = name
		}
		visible = {
			international_organization_type = international_organization_type:ilkhanate
		}
	}

	effect = {
		if = {
			limit = { exists = scope:actor }
			scope:actor = {
				add_government_power = government_power_mild_penalty
				international_organization:ilkhanate = {
					international_organization_add_special_status = {
						type = special_status:ilkhan_claimant
						country = scope:actor
					}
				}
			}
		}
	}
	ai_will_do = {
		add = 10 #Base value

		scope:actor = {
			subtract = {
				desc = "game_concept_government_power"
				value = "currency_utility(government_power|government_power_mild_penalty)"
			}

			ruler ?= {
				desc = "RULER_ABILITY"
				add = total_abilities
				multiply = 0.02
			}

			heir ?= {
				desc = "HEIR_ABILITY"
				add = total_abilities
				multiply = 0.02
			}

			add = {
				desc = "SIZE_OF_COUNTRY_TOOLTIP"
				value = total_population
				add = num_locations
				divide = 100
			}

			add = {
				desc = "game_concept_army"
				value = army_size
				multiply = 5
			}
		}

		#Subtract the power of other claimants
		international_organization:ilkhanate = {
			every_international_organization_member = {
				limit = {
					this != scope:actor
					has_special_status_in_international_organization = {
						type = special_status:ilkhan_claimant
						international_organization = international_organization:ilkhanate
					}
				}

				add = {
					desc = "SIZE_OF_COUNTRY_TOOLTIP"
					value = total_population
					add = num_locations
					divide = -100
				}
	
				add = {
					desc = "game_concept_army"
					value = army_size
					multiply = -5
				}
	
			}
		}
	}
}
