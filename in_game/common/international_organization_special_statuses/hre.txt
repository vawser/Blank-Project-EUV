free_city = {
	priority = 10

	modifier = {
		burghers_estate_target_satisfaction = small_permanent_target_satisfaction
		global_monthly_development = 0.001
		global_production_efficiency = medium_production_efficiency_bonus
		reject_subjugation_reasons = 50
	}

	can_bestow_trigger = {
		is_subject = no
		num_of_non_rural = 1
		num_locations = 1
		is_member_of_international_organization = scope:recipient
		scope:recipient = {
			hre_enabled_free_cities = yes
			leader_country ?= {
				religion = root.religion
			}
			NOT = {
				country_has_special_status = { type = special_status:free_city country = root }
			}
			NOT = {
				country_has_special_status = { type = special_status:elector country = root }
			}
			international_organization_has_law = law:free_city_rights_law
			OR = {
				AND = {
					international_organization_has_policy = policy:german_free_cities_only_policy
					root.culture = { has_culture_group = culture_group:german_group }
				}
				AND = {
					international_organization_has_policy = policy:german_and_german_adjacent_free_cities_policy
					root.culture = {
						OR = {
							has_culture_group = culture_group:german_group
							has_culture_group = culture_group:netherlandish_group
							has_culture_group = culture_group:swiss_group
							has_culture_group = culture_group:italian_group
						}
					}
				}
				international_organization_has_policy = policy:empire_wide_free_cities_policy
			}
		}
	}
	
	auto_bestowal_trigger = {
	}

	auto_rescind_trigger = {
		OR = {
			num_of_non_rural > 1
			num_locations = 0
			NOT = { is_member_of_international_organization = scope:recipient }
			scope:recipient = {
				hre_disabled_free_cities = yes
			}
		}
	}
	
	on_bestowed_effect = {
		trigger_event_silently = free_cities.1
		subject_type:imperial_free_city = { save_scope_as = type_scope }
		if = {
			limit = { scope:recipient = { hre_has_direct_free_city_subjects = yes } }
			subject_type:direct_imperial_free_city = { save_scope_as = type_scope }
		}
		if = {
			limit = { exists = scope:type_scope }
			hre_convert_free_city_to_emperor_subject = {
				international_organization = scope:recipient
				target = root
				type = scope:type_scope
			}
		}
	}
	
	on_rescinded_effect = {
		trigger_event_silently = free_cities.2
		hre_break_free_city_emperor_subject = yes
	}

	map_color = define:NMapColors|HRE_FREE_CITY_COLOR
	special_status_power = {
		value = 25
		if = {
			limit = { 
				OR = {
					scope:recipient = { international_organization_has_policy = policy:no_imperial_diet_policy } 
					scope:recipient = { international_organization_has_policy = policy:no_golden_bull_policy } 
				}
			}
			multiply = 0.1
		}
		else_if = {
			limit = { scope:recipient = { international_organization_has_policy = policy:hre_early_imperial_diet_policy } }
			multiply = 0.25
		}
		else_if = {
			limit = { scope:recipient = { international_organization_has_policy = policy:hre_bi_camerial_imperial_diet_policy } }
			multiply = 0.0
		}
		else_if = {
			limit = { scope:recipient = { international_organization_has_policy = policy:hre_tri_camerial_imperial_diet_policy } }
			multiply = 2
		}
	}
}
