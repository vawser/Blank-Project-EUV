japanese_emperor = {
	priority = 10

	modifier = {
	}

	can_bestow_trigger = {
		is_subject = no
		has_reform = government_reform:japanese_imperial_family
		is_member_of_international_organization = scope:recipient
		scope:recipient = {
			NOT = {
				country_has_special_status = {
					type = scope:source
					country = root
				}
			}
		}
	}

	auto_bestowal_trigger = {
		always = yes
	}

	auto_rescind_trigger = {
		OR = {
			NOT = { is_member_of_international_organization = scope:recipient }
			NOT = { has_reform = government_reform:japanese_imperial_family }
		}
	}
	
	on_bestowed_effect = {
	}
	
	on_rescinded_effect = {
	}

	map_color = define:NMapColors|INTERNATIONAL_ORGANIZATION_LEADER_COLOR
}

shugo_daimyo = {
	priority = 10

	can_bestow_trigger = {
		always = no
	}

	auto_bestowal_trigger = {
		country_type = building
		any_owned_foreign_building = {
			building_type = building_type:kokufu
		}
	}

	auto_rescind_trigger = {
		OR = {
			NOT = { is_member_of_international_organization = scope:recipient }
			NOT = {
				any_owned_foreign_building = {
					building_type = building_type:kokufu
				}
			}
		}
	}
	
	on_bestowed_effect = {
	}
	
	on_rescinded_effect = {
	}

	modifier = {
		country_cabinet_efficiency = 0.1
		monthly_legitimacy = 0.1
		tax_income_efficiency = large_tax_income_efficiency_bonus
	}

	leader_modifier = {
		country_cabinet_efficiency = 0.005
	}

	map_color = define:NMapColors|INTERNATIONAL_ORGANIZATION_TIER1_COLOR
}
