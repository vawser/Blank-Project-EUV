curia = {
	priority = 10

	modifier = {
	}

	can_bestow_trigger = {
	}

	auto_bestowal_trigger = {
		num_cardinals > 0
	}

	auto_rescind_trigger = {
		num_cardinals = 0
	}
	
	on_bestowed_effect = {
	}
	
	on_rescinded_effect = {
	}
	map_color = {
		if = {
			limit = {
				root.owner = {
					modifier:is_pope = yes
			}
			}
			value = define:NMapColors|INTERNATIONAL_ORGANIZATION_LEADER_COLOR
		}
		else = {
			value = define:NMapColors|INTERNATIONAL_ORGANIZATION_TIER1_COLOR
		}
	}
}
