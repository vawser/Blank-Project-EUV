ilkhan_claimant = {
	priority = 10

	modifier = {
	}

	can_bestow_trigger = {
		always = no #it's not granted, it's claimed
	}

	auto_bestowal_trigger = {
	}

	auto_rescind_trigger = {
	}
	
	on_bestowed_effect = {
	}
	
	on_rescinded_effect = {
		if = {
			limit = {
				scope:recipient = {
					num_countries_with_special_status = {
						type = scope:source
						value = 1
					}
				}
			}
			scope:recipient = {
				every_international_organization_member = {
					limit = {
						has_special_status_in_international_organization = {
							type = special_status:ilkhan_claimant
							international_organization = scope:recipient
						}
					}
					scope:recipient = { set_leader_country = prev }
				}
			}
		}
	}

	map_color = {
		if = {
			limit = {
				root.owner = {
					is_leader_of_international_organization = international_organization:ilkhanate
			}
			}
			value = define:NMapColors|INTERNATIONAL_ORGANIZATION_LEADER_COLOR
		}
		else = {
			value = define:NMapColors|INTERNATIONAL_ORGANIZATION_TIER1_COLOR
		}
	}
}
