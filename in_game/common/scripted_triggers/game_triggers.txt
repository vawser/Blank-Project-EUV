game_is_playing_on_easy_mode = {
	OR = {
		has_game_rule = player_very_easy_difficulty
		has_game_rule = player_easy_difficulty
	}
	OR = {
		has_game_rule = ai_very_easy_difficulty
		has_game_rule = ai_easy_difficulty
	}
}

game_is_playing_on_hard_mode = {
	NOR = { 
		has_game_rule = player_very_easy_difficulty
		has_game_rule = player_easy_difficulty
		has_game_rule = player_normal_difficulty
		has_game_rule = ai_very_easy_difficulty
		has_game_rule = ai_easy_difficulty
		has_game_rule = ai_normal_difficulty
	}
	OR = {
		AND = {
			has_game_rule = player_normal_difficulty
			NOT = { has_game_rule = ai_normal_difficulty }
		}
		AND = {
			has_game_rule = ai_normal_difficulty
			NOT = { has_game_rule = player_normal_difficulty }
		}
		AND = {
			NOT = { has_game_rule = player_normal_difficulty }
			NOT = { has_game_rule = ai_normal_difficulty }
		}
	}
}

game_has_missions_enabled = {
	NOT = { has_game_rule = mission_packs_disabled }
	#General check for all types of mission
	#Using this formatting in case we add additional types of missions which should be en- and disabled in the settings
}

game_has_mission_rewards_enabled = {
	OR = {
		has_game_rule = every_task_reward
		has_game_rule = only_end_mission_rewards
	}
}

game_has_mission_task_rewards_enabled = {
	has_game_rule = every_task_reward
}

game_has_victory_cards_enabled = {
	has_game_rule = victory_cards_enabled
}

game_is_initialized = {
	current_date > 1337.5.1
}

current_later_age_1_traditions_trigger = {
	always = yes	#it's always age 1 or later
}
current_later_age_2_renaissance_trigger = {
	custom_tooltip = {
		text = current_age_at_least_2_tt
		NOT = { current_age = age_1_traditions }
	}
}

current_later_age_3_discovery_trigger = {
	custom_tooltip = {

		NOT = { current_age = age_1_traditions }
		NOT = { current_age = age_2_renaissance }
	}
}

current_later_age_4_reformation_trigger = {
	custom_tooltip = {
		text = current_age_at_least_4_tt
		NOT = { current_age = age_1_traditions }
		NOT = { current_age = age_2_renaissance }

current_later_age_5_absolutism_trigger = {
	custom_tooltip = {
		text = current_age_at_least_5_tt
		OR = {

		}
	}
}

	}

		}
