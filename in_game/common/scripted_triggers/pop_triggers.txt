is_dhimmi = {
	OR = {	
		religion.group = religion_group:buddhist
		religion.group = religion_group:christian
		religion.group = religion_group:dharmic
		religion.group = religion_group:israelite_group
		religion.group = religion_group:mandean_group
		religion.group = religion_group:manichaean_group
		religion.group = religion_group:zoroastrian_group
	}
	owner ?= { 
		religion.group = religion_group:muslim
	}
	is_linked_to_foreign_building = no
}

is_cossacks = {
	culture = culture:cossack_culture
	owner ?= {
		OR = {
			culture = { has_culture_group = culture_group:russian_group }
			culture = { has_culture_group = culture_group:lithuanian_group }
			culture = { has_culture_group = culture_group:polish_group }
		}
	}
}

is_gaelic_clans = {
	culture ?= {
		language = language:gaelic_language
		has_culture_group = culture_group:celtic_group
	}
	owner ?= {
		has_estate_privilege = estate_privilege:scottish_clans
	}
	is_linked_to_foreign_building = no
}

is_not_protected_from_conversion = {
	OR = {
		AND = {
			owner ?= { modifier:global_nobles_conversion_blocked = no }
			location = { modifier:local_nobles_conversion_blocked = no }
			pop_type = pop_type:nobles
		}
		AND = {
			owner ?= { modifier:global_clergy_conversion_blocked = no }
			location = { modifier:local_clergy_conversion_blocked = no }
			pop_type = pop_type:clergy
		}
		AND = {
			owner ?= { modifier:global_burghers_conversion_blocked = no }
			location = { modifier:local_burghers_conversion_blocked = no }
			pop_type = pop_type:burghers
		}
		AND = {
			owner ?= { modifier:global_peasants_conversion_blocked = no }
			location = { modifier:local_peasants_conversion_blocked = no }
			pop_type = pop_type:peasants
		}
		AND = {
			owner ?= { modifier:global_slaves_conversion_blocked = no }
			location = { modifier:local_slaves_conversion_blocked = no }
			pop_type = pop_type:slaves
		}
		AND = {
			owner ?= { modifier:global_tribesmen_conversion_blocked = no }
			location = { modifier:local_tribesmen_conversion_blocked = no }
			pop_type = pop_type:tribesmen
		}
		AND = {
			owner ?= { modifier:global_laborers_conversion_blocked = no }
			location = { modifier:local_laborers_conversion_blocked = no }
			pop_type = pop_type:laborers
		}
		AND = {
			owner ?= { modifier:global_soldiers_conversion_blocked = no }
			location = { modifier:local_soldiers_conversion_blocked = no }
			pop_type = pop_type:soldiers
		}
	}
}
