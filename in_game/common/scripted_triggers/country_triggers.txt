country_rank_is_what = {
	trigger_if = {
		limit = {
			exists = scope:character
			has_ruler = yes
			not = { ruler = scope:character }
			scope:character ?= {
				exists = "rule_end_date(root)"
			}
		}
		"country_rank_on_date(scope:character.rule_end_date(root))" ?= $target_rank$
	}
	trigger_else = {
		country_rank ?= $target_rank$
	}
}

	is_great_power = yes

		limit = { 
			country_exists = c:PAP

			opinion = { target = root value > 100 }

alliance_can_exist = {
	scope:actor = {
		opinion = { target = scope:recipient value >= 0 }
		not = { has_blocked_treaties = scope:recipient }
		not = { is_subject_of = scope:recipient }
		not = { is_rival_of = scope:recipient }
		modifier:allowed_alliance = yes
		NOT = {
			is_member_of_international_organization_of_type = {
				type = coalition
				target = scope:recipient
			}
		}
		NOT = {
			in_union_with = scope:recipient
		}
		is_rebel_country = no
	}
	scope:recipient = {
		opinion = { target = scope:actor value >= 0 }
		not = { has_blocked_treaties = scope:actor }
		not = { is_subject_of = scope:actor }
		not = { is_rival_of = scope:actor }
		modifier:allowed_alliance = yes
		NOT = {
			is_member_of_international_organization_of_type = {
				type = coalition
				target = scope:actor
			}
		}
		is_rebel_country = no
	}
	can_create_subject_alliance = {
		actor = scope:actor
		recipient = scope:recipient
	}
	not = {
		and = {
			scope:actor = {
				is_hegemon = yes
			}
			scope:recipient = {
				is_hegemon = yes
			}
		}
	}
}

can_change_primary_culture = {
	$country$ = {
		trigger_if = {
			limit = {
				is_subject = yes
			}
			trigger_if = {
				limit = {
					subject_type ?= { only_allowed_overlord_primary_culture = yes }
				}
				overlord ?= {
					culture = $culture$
				}
			}
			trigger_else_if = {
				limit = {
					subject_type ?= { only_allowed_overlord_primary_or_kindred_culture = yes }
				}
				overlord ?= {
					culture ?= {
						OR = {
							this = $culture$
							cultural_view = {
								target = $culture$
								value = kindred
							}
						}
					}
				}
			}
			trigger_else = {
				hidden_trigger = { always = yes }
			}
		}
	}
}

is_eligible_for_the_great_ruler_event = {
	has_ruler = yes
	ruler ?= {
		total_abilities >= 250
	}
	is_subject = no
	ruler_reign >= 5
	has_or_had_great_ruler = no	#There has never been any "the Great" before (historical "the Great"s excluded)
	average_control_in_home_region >= 0.33
	OR = {
		country_rank_level >= 3
		is_great_power = yes
	}
	stability >= 75
	has_any_active_disaster = no
	prestige >= 75
	is_during_bankruptcy = no
	government_power >= 90
	war_exhaustion < 2
	num_works_of_art >= 4
	monthly_balance > 0
	OR = {
		at_war = no
		#Is Winning against a significant power
		any_great_power = {
			is_at_war_with = ROOT
			ROOT = {
				war_score_versus = {
					target = PREV
					value >= 0.4
				}
			}
		}
	}
}

can_bribe_default_trigger = {
	exists = union
	exists = scope:recipient
	union ?= {
		modifier:union_integration_level > 0
	}
	trigger_if = {
		limit = { scope:recipient = { has_variable = integrated_in_union } }
		save_temporary_scope_value_as = {
			name = union_cost_factor_multiplier
			value = {
				value = union.modifier:union_integration_level
				subtract = scope:recipient.var:integrated_in_union
			}
		}
	}
	trigger_if = {
		limit = {
			great_power_score > 0
			scope:recipient.great_power_score > 0
		}
		save_temporary_scope_value_as = {
			name = union_special_status_multiplier
			value = {
				value = great_power_score
				divide = scope:recipient.great_power_score
				add = 1
			}
		}
	}
	trigger_else = {
		save_temporary_scope_value_as = {
			name = union_cost_factor_multiplier
			value = union.modifier:union_integration_level
		}
	}
	trigger_if = {
		limit = { exists = scope:union_cost_factor_multiplier }
		government_power >= {
			value = government_power_weak_penalty
			multiply = scope:union_cost_factor_multiplier
			if = {
				limit = { exists = scope:union_special_status_multiplier }
				multiply = scope:union_special_status_multiplier
			}
			max = 50
		}
		gold >= union_parliament_gold_bribe_recipient
	}
}
