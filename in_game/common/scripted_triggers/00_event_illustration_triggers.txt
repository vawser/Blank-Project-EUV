event_illustration_background_trigger = {
	
	# location
	trigger_if = {
		limit = { NOT = { exists = scope:target_location } }

		trigger_if = { limit = { exists = scope:target_pop_bckg } scope:target_pop_bckg.location = { save_temporary_scope_as = target_location } }

		trigger_if = {
			limit = { NOT = { exists = scope:target_location } }			
			trigger_if = { limit = { exists = scope:target_character } scope:target_character.location = { save_temporary_scope_as = target_location } }
		}
		trigger_if = {
			limit = { NOT = { exists = scope:target_location } }			
			trigger_if = { limit = { exists = capital } capital = { save_temporary_scope_as = target_location } }
		}
	}

	# culture
	trigger_if = {
		limit = { NOT = { exists = scope:target_culture_bckg } }

		trigger_if = { limit = { exists = scope:target_pop_bckg } scope:target_pop_bckg.culture = { save_temporary_scope_as = target_culture_bckg } }

		trigger_if = {
			limit = { NOT = { exists = scope:target_culture_bckg } }
			trigger_if = { limit = { exists = scope:target_location } scope:target_location.dominant_culture = { save_temporary_scope_as = target_culture_bckg } }
		}
		trigger_if = {
			limit = { NOT = { exists = scope:target_culture_bckg } }
			root.culture = { save_temporary_scope_as = target_culture_bckg }
		}
	}

	# religion
	trigger_if = {
		limit = { NOT = { exists = scope:target_religion_bckg } }

		trigger_if = { limit = { exists = scope:target_pop_bckg } scope:target_pop_bckg.religion = { save_temporary_scope_as = target_religion_bckg } }

		trigger_if = {
			limit = { NOT = { exists = scope:target_religion_bckg } }
			trigger_if = { limit = { exists = scope:target_location } scope:target_location.dominant_religion = { save_temporary_scope_as = target_religion_bckg } }
		}
		trigger_if = {
			limit = { NOT = { exists = scope:target_religion_bckg } }
			root.religion = { save_temporary_scope_as = target_religion_bckg }
		}
	}

	# pop type 
	trigger_if = {
		limit = { NOT = { exists = scope:target_pop_type_bckg } }

		trigger_if = { limit = { exists = scope:target_pop_bckg } scope:target_pop_bckg.pop_type = { save_temporary_scope_as = target_pop_type_bckg } }
	}
}
