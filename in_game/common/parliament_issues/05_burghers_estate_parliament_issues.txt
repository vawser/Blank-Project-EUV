
sponsor_a_university = {
	estate = burghers_estate

	modifier_when_in_debate = {
		research_speed_modifier = 0.05
		monthly_towards_innovative = societal_value_huge_monthly_move #moves value by 3 over the duration
	}

	allow = {
		has_advance = university_advance
		NOT = {
			has_estate_privilege = estate_privilege:clergy_lecture_oaths
		}
		any_owned_location = {
			is_core_of = root
			OR = {
				location_rank = location_rank:town
				location_rank = location_rank:city
			}
			NOT = { has_building = building_type:university }
		}
	}

	chance = {
		add = 1
		add = {
			if = {
				limit = {
					societal_value:traditionalist_vs_innovative > 30
				}
				value = 2
			}
		}
		
		add = {
			if = {
				limit = {
					total_effective_building_levels:university < 1
				}
				value = 2
			}
		}
		multiply = {
			if = {
				limit = {
					societal_value:traditionalist_vs_innovative < 0
				}
				value = 0.5
			}
			else = {
				value = 1	#We need that else or otherwise the whole chance is multiplied with 0
			}
		}
	}
	
	on_debate_passed = {
		ordered_owned_location = {
			order_by = {
				value = population
				if = {
					limit = { location_rank = location_rank:city }
					multiply = 5
				}
			}
			limit = {
				is_core_of = root
				NOT = { has_building = building_type:university }
				OR = {
					location_rank = location_rank:town
					location_rank = location_rank:city
				}
			}
			max = 1
			construct_building = {
				building_type = building_type:university
				cost_multiplier = 0.5
				cost_multiplier_reason = "game_concept_parliament_issue"
			}
			
		}
	}
	on_debate_failed = {
		parliament_debate_failed_effect = yes
	}
}

promote_urbanization = {
	estate = burghers_estate

	modifier_when_in_debate = {
		global_pop_promotion_speed_modifier = 1
	}

	allow = {
		OR = {
			current_age = age_1_traditions
			current_age = age_2_renaissance
			current_age = age_3_discovery
			current_age = age_4_reformation
		}
		at_war = no
		modifier:can_build_cities = yes
		num_locations > 1
		any_owned_rural_location = {
			NOT = {
				any_neighbor_location = {
					is_city = yes
				}
			}
		}
	}

	chance = {
		add = 1
		add = {
			if = {
				limit = {
					num_locations > 5
				}
				value = 2
			}
		}

		add = {
			if = {
				limit = {
					num_of_non_rural = 0
				}
				value = 4
			}
		}
		
		multiply = {
			if = {
				limit = {
					num_of_non_rural > 5
				}
				value = 0.5
			}
			else = {
				value = 1	#We need that else or otherwise the whole chance is multiplied with 0
			}
		}
	}
	
	on_debate_passed = {
		ordered_owned_rural_location = {
			limit = {
				NOT = {
					any_neighbor_location = {
						is_city = yes
					}
				}
			}
			order_by = {
				value = development
				add = population
				add = num_buildings
			}
			max = 1
			change_location_rank = location_rank:town
			every_pop = {
				limit = {
					owner = root
				}
				add_pop_satisfaction = pop_satisfaction_ultimate_bonus 
			}
			
		}
	}
	on_debate_failed = {
		parliament_debate_failed_effect = yes
	}
}

increase_naval_spending = {
	estate = burghers_estate

	modifier_when_in_debate = {
		ship_build_speed = 0.25
	}

	allow = {
		num_of_ports >= 1
	}

	chance = {
		add = 1
		add = {
			if = {
				limit = {
					societal_value:land_vs_naval > 30
				}
				value = 2
			}
		}
		add = {
			if = {
				limit = {
					navy_size_percentage <= 0.5
				}
				value = 2
			}
		}
		multiply = {
			if = {
				limit = {
					societal_value:land_vs_naval < -30
				}
				value = 0.5
			}
			else = {
				value = 1	#We need that else or otherwise the whole chance is multiplied with 0
			}
		}
	}
	
	on_debate_passed = {
		if = {
			limit = {
				num_of_ports <= 10
			}
			ordered_port_in_country = {
				order_by = population
				check_range_bounds = no
				max = {
					value = root.num_of_ports
					multiply = 0.5
					add = 1
				}
				change_development  = development_weak_bonus
			}
		}
		else_if = {
			limit = {
				num_of_ports > 10
				num_of_ports <= 25
			}
			ordered_port_in_country = {
				order_by = population
				check_range_bounds = no
				max = {
					value = root.num_of_ports
					multiply = 0.25
					add = 1
				}
				change_development  = development_weak_bonus
			}
		}
		else_if = {
			limit = {
				num_of_ports > 25
			}
			ordered_port_in_country = {
				order_by = population
				check_range_bounds = no
				max = {
					value = root.num_of_ports
					multiply = 0.1
					add = 1
				}
				change_development  = development_weak_bonus
			}
		}
		if = {
			limit = {
				has_variable = bng_buffed_naval_parliament_debate_flag
			}
			change_societal_value = {
				type = land_vs_naval
				value = societal_value_minor_move_to_right
			}
		}
		else = {
			change_societal_value = {
				type = land_vs_naval
				value = societal_value_tiny_move_to_right
			}
		}
		
	}
	on_debate_failed = {
		parliament_debate_failed_effect = yes
		change_societal_value = {
			type = land_vs_naval
			value = societal_value_tiny_move_to_left
		}
	}
}
