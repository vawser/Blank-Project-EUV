italian_wars_land_ownership = {

	# applied to locations that are owned by the international organization
	modifier = {
	}

	can_add_trigger = {
		always = yes
	}

	# root = location, recipient = international organization
	can_add_location_trigger = {
		is_land = yes
		is_passable = yes
		region = region:italy_region
	}
	
	# root = location, recipient = international organization
	can_remove_trigger = {
		always = no
	}
	
	# root = location, recipient = international organization
	on_added = {
		scope:recipient.leader_country ?= {
			if = {
				limit = {
					NOT = {
						this = root.owner
					}
				}
				add_opinion = { target = root.owner modifier = opinion_strengthened_our_league }
			}
			
		}
	}
	
	# root = location, recipient = international organization
	on_removed = {
		scope:recipient.leader_country ?= {
			if = {
				limit = {
					NOT = {
						this = root.owner
					}
				}
				add_opinion = { target = root.owner modifier = opinion_weakened_our_league }
			}
		}
	}
	
	# root = location, recipient = international organization
	ai_desire_to_add = {	#You should always want to add the location to the IO
		add = 20
	}

	owned_location_color = rgb { 0 59 70 }
}
