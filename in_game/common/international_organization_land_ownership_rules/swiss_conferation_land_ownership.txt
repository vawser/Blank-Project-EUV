swiss_confederation_land_ownership = {

	# applied to locations that are owned by the international organization
	modifier = {
	}
	
	can_add_trigger = {
		always = yes
	}
	can_remove_trigger = {
		always = yes
	}

	# root = location, recipient = international organization
	can_add_location_trigger = {
		is_land = yes
		is_passable = yes
		OR = {
			area = area:central_switzerland_area
			area = area:eastern_switzerland_area
			area = area:western_switzerland_area
			any_neighbor_location = {
				is_owned_by_international_organization = scope:recipient
			}
		}
	}
	
	# root = location, recipient = international organization
	can_remove_location_trigger = {
		NOT = { area = area:central_switzerland_area }
		NOT = { area = area:eastern_switzerland_area }
		NOT = { area = area:western_switzerland_area }
		owner ?= {
			NOT = { is_leader_of_international_organization = international_organization:swiss_confederation }
		}
	}
	
	# root = location, recipient = international organization
	on_added = {
	}
	
	# root = location, recipient = international organization
	on_removed = {
	}
	
	# root = location, recipient = international organization
	ai_desire_to_add = {
	}

	#owned_location_color = rgb { 155 75 190 }
	
	allow_control_propagation = yes
}
