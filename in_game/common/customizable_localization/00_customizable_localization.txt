#Make sure to use set_character_siblings_scope in an immediate beforehand, or else you always get "siblings"
GetSibling = {
	type = country
	text = {
		localization_key = family_relations_sister
		trigger = {
			exists = scope:first_sibling
			exists = scope:second_sibling
			scope:first_sibling ?= { is_female = yes }
			scope:second_sibling ?= { is_female = yes }
		}
	}
	text = {
		localization_key = family_relations_brother
		trigger = {
			exists = scope:first_sibling
			exists = scope:second_sibling
			scope:first_sibling ?= { is_female = no }
			scope:second_sibling ?= { is_female = no }
		}
	}
	text = {
		localization_key = family_relations_sibling
		fallback = yes
	}
}
