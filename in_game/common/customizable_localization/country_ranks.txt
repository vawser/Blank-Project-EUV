country_flavor_heir = {
	type = country

	text = {
		localization_key = country_flavor_heir_successor
		trigger = {
			OR = {
				government_type = government_type:theocracy
				government_type = government_type:republic
			}
		}
	}

	text = {
		localization_key = country_flavor_heir_tanist
		trigger = {
			succession_law = heir_selection:tanistry_elective
		}
	}

	text = {
		localization_key = country_flavor_heir_crown_prince
		trigger = {
			country_rank_level >= 3 # kingdom
			government_type = government_type:monarchy
		}
	}

	text = {
		localization_key = country_flavor_heir
		fallback = yes
	}
}
