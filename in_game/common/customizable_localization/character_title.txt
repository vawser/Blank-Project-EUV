character_title_prefix = {
	type = character

	text = {
		localization_key = character_title_guru

		trigger = {
			exists = religion
			religion ?= {
				exists = group
				group ?= religion_group:dharmic
			}
			OR = {
				is_saint = yes
				is_religious_figure = yes
			}
		}
	}

	text = {
		localization_key = character_title_saint

		trigger = {
			is_saint = yes
		}
	}

	text = {
		localization_key = character_title_doctor

		trigger = {
			is_ruler = no
			is_legally_male = yes # very unlikely (but not impossible) for women to become doctors
			age_in_years >= 25
			has_estate = estate_type:clergy_estate
			adm >= 67
			exists = birth_location
			birth_location = { has_building_with_at_least_one_level = university }
			religion.group = religion_group:christian
		}
	}

	text = {
		localization_key = character_title_nobility_pirate

		trigger = {
			is_adult = yes
			is_ruler = no
			OR = {
				is_mercenary_leader = yes
				AND = {
					has_estate = estate_type:nobles_estate
					owner ?= {
						has_reform = government_reform:pirate_brethren_reform
					}
				}
			}
		}
	}

	text = {
		localization_key = character_title_nobility_french

		trigger = {
			is_adult = yes
			is_ruler = no
			OR = {
				has_estate = estate_type:nobles_estate
				AND = {
					has_estate = estate_type:crown_estate
					owner ?= { government_type = government_type:monarchy }
				}
			}
			exists = culture
			exists = culture.language
			culture.language = language:french_language

			owner ?= {
				OR = {
					# not really used in french
					culture.language = language:english_language
					court_language ?= language:english_language
				}
			}
		}
	}

	text = {
		localization_key = character_title_nobility_welsh

		trigger = {
			is_adult = yes
			is_ruler = no
			OR = {
				has_estate = estate_type:nobles_estate
				AND = {
					has_estate = estate_type:crown_estate
					owner ?= { government_type = government_type:monarchy }
				}
			}
			exists = culture
			exists = culture.language
			culture.dialect = dialect:welsh_dialect

			owner ?= {
				OR = {
					# not really used in welsh
					culture.language = language:english_language
					court_language ?= language:english_language
				}
			}
		}
	}

	text = {
		localization_key = character_title_nobility_english

		trigger = {
			is_adult = yes
			is_ruler = no
			OR = {
				has_estate = estate_type:nobles_estate
				AND = {
					has_estate = estate_type:crown_estate
					owner ?= { government_type = government_type:monarchy }
				}
			}

			OR = {
				AND = {
					exists = culture
					exists = culture.language
					culture.language = language:english_language
				}
				owner ?= {
					OR = {
						AND = {
							exists = culture
							exists = culture.language
							culture.language = language:english_language
						}
						court_language ?= language:english_language
					}
				}
			}
		}
	}

	text = {
		localization_key = character_title_nobility_malay

		trigger = {
			is_ruler = no

			OR = {
				has_estate = estate_type:nobles_estate
				AND = {
					has_estate = estate_type:crown_estate
					owner ?= { government_type = government_type:monarchy }
				}
			}

			OR = {
				AND = {
					exists = culture
					exists = culture.language
					culture.language = language:malay_language
				}
				owner ?= {
					OR = {
						AND = {
							exists = culture
							exists = culture.language
							culture.language = language:malay_language
						}
						court_language ?= language:malay_language
					}
				}
			}
		}
	}
}
character_title_prefix_female = { parent = character_title_prefix suffix = "_female" }

character_title_suffix = {
	type = character

	text = {
		localization_key = character_title_shinno

		trigger = {
			is_ruler = no
			exists = culture
			exists = culture.language
			culture.language = language:japanese_language
			dynasty ?= dynasty:yamato_dynasty
			owner ?= {
				OR = {
					AND = {
						exists = culture
						exists = culture.language
						culture.language = language:japanese_language
					}
					court_language ?= language:japanese_language
				}
			}
		}
	}

	text = {
		localization_key = character_title_minister_of_parliament

		trigger = {
			in_cabinet = yes
			is_ruler = no

			owner ?= {
				parliament_type = parliament_type:constitutional_parliament
				OR = {
					culture.language = language:english_language
					court_language ?= language:english_language
				}
			}
		}
	}

	text = {
		localization_key = character_title_burgher_esquire

		trigger = {
			is_adult = yes
			is_female = no
			is_ruler = no
			has_estate = estate_type:burghers_estate
			adm > 75

			OR = {
				AND = {
					exists = culture
					exists = culture.language
					culture.language = language:english_language
				}
				owner ?= {
					OR = {
						AND = {
							exists = culture
							exists = culture.language
							culture.language = language:english_language
						}
						court_language ?= language:english_language
					}
				}
			}
		}
	}
}
character_title_suffix_female = { parent = character_title_suffix suffix = "_female" }
