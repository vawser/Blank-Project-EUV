	locked = {
		custom_tooltip = {
			text = fundamental_law_tt
			OR = {
				AND = {
					var:religion ?= religion:orthodox
					international_organization_has_policy = policy:dual_nature
				}
				AND = {
					var:religion ?= religion:miaphysite
					international_organization_has_policy = policy:composite_nature
				}
				AND = {
					var:religion ?= religion:nestorianism
					international_organization_has_policy = policy:separate_nature
				}
			}
		}
	}

	single_nature = {
		years = 1
		price = christian_tenet_price
		allow = {
			var:religion ?= {
				NOR = {
					this = religion:orthodox
					this = religion:miaphysite
					this = religion:nestorianism
				}
			}
		}
		on_activate = {
			custom_tooltip = {
				text = change_maintenance_tt
				every_international_organization_member = {
					set_variable = enable_single_nature_pm_variable
					if = {
						limit = { has_variable = enable_composite_nature_pm_variable }
						remove_variable = enable_composite_nature_pm_variable
					}
					if = {
						limit = { has_variable = enable_dual_nature_pm_variable }
						remove_variable = enable_dual_nature_pm_variable
					}
					if = {
						limit = { has_variable = enable_separate_nature_pm_variable }
						remove_variable = enable_separate_nature_pm_variable
					}
				}
			}
		}
		country_modifier = {
			#leader benefits
			potential_trigger = {
				is_leader_of_international_organization_of_type = {
					type = autocephalous_patriarchate
				}
			}
