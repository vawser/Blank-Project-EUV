piracy_law = {
	law_category = military
	potential = {
		num_of_ports > 0
	}

	state_piracy_policy = {
		allow = {
			custom_tooltip = {
				text = has_state_piracy_policy_access
				OR = {
					has_unlocked_policy_trigger = { type = state_piracy_policy }
					tag = KNI
					tag = TUR 
					has_advance = privateers
					has_advance = letters_of_marque
					has_reform = government_reform:galley_lords_reform
					has_reform = government_reform:pirate_brethren_reform
					capital.region = region:maghreb_region
				}
			}
		}
		country_modifier = {
			can_hire_privateers = yes
			privateer_maintenance_cost_modifier = -0.33
			can_be_target_of_anti_piracy_cb = yes
		}
		estate_preferences = {
			nobles_estate
		}
		years = 2
	}	
	issue_letter_of_marque_policy = {
		allow = {
			custom_tooltip = {
				text = has_privateers_or_LoM_advances
				OR = {
					has_advance = privateers
					has_advance = letters_of_marque
				}
			}
		}
		country_modifier = {
			hire_privateer_cost_modifier = -0.25
			navy_maintenance_cost = 0.10
		}
		estate_preferences = {
			crown_estate
		}
		years = 2
	}
	
	anti_piracy_policy = {
		allow = {
			NOR = {
				has_reform = government_reform:galley_lords_reform
				has_reform = government_reform:pirate_brethren_reform
			}
		}
		country_modifier = {
			anti_piracy_warfare_modifier = 0.25
			can_create_anti_piracy_cb = yes
		}
		estate_preferences = {
			burghers_estate
		}
		years = 2
		on_activate = {
			custom_tooltip = {
				text = anti_piracy_policy_at
				every_known_country = {
					limit = {
						has_policy = state_piracy_policy 
					}
					add_opinion = {
						modifier = allows_state_piracy_modifier
						target = ROOT
					}
				}
			}
		}
		on_deactivate = {
			custom_tooltip = {
				text = anti_piracy_policy_dt
				every_known_country = {
					limit = {
						has_opinion = { 
							target = ROOT 
							modifier = allows_state_piracy_modifier 
						}
					}
					remove_opinion = { 
						target = ROOT
						modifier = allows_state_piracy_modifier 
					}
				}
			}
		}
	}
	sea_beggars_policy = {
