#Intro:
#The majority of HRE laws have 5 policies with a very few which have less like the Golden Bull.
#With the enactment of the Golden Bull, most HRE laws are set to the 3rd policy which has limited or no modifiers active at all
#The HRE Emperor can try to change the policy to the next adjacent one (so from 3rd tier to 4th or 2nd tier)
#The left tier is the 4th and 5th and is more in favor of benefitting the HRE as a whole
#The right tier is the 2nd and 1st and is more in favor of benefitting the HRE Emperor first and foremost

#Some laws have only 3 policies in which cases there are not pro-Emperor vs pro-Member positions

#The only law which does not follow the rule is the power_of_the_emperor_law which starts at improved_imperial_authority_policy (4) policy.
#The reason for this is to delay the unification of the HRE + there is not much design space for an even further decentralized HRE.
#As such, having 2 policies for the decentralized HRE are not really doable.

#								LAW
# Pol 5 		 Pol 4			Pol 3		   Pol 2		  Pol 1
#   |--------------|--------------|--------------|--------------|

	type = international_organization
	potential = { international_organization_type = international_organization_type:hre }
	allow = {
		has_currency_to_vote_for_law = { value = imperial_authority_uncontroversial_law }
	}
	locked = {
		custom_description = {
			text = hre_has_locked_law
			subject = law:golden_bull
			hre_has_locked_law = { law = golden_bull }
		}
	}
	custom_tags = { forbids_no_policy }

	golden_bull_policy = {
		# How much the Emperor wants to propose a policy
		wants_propose_policy = {
			add = {
				desc = "POLICY_ESSENTIAL_REFORM"
				value = 1000
			}
			add = hre_default_emperor_policy_desire
		}
		# How much the princes want to vote for / against the policy
		wants_this_policy_bias = {
			add = {
				desc = "POLICY_ESSENTIAL_REFORM"
				value = 500 #Literally everyone and their grandmother wants this reform to pass
			}
			add = hre_default_policy_opinion
		}
		on_activate = {
			hre_lock_law = { law = golden_bull }
			hre_setup_hre_variables = yes
			hre_enable_all_imperial_laws = yes
			hre_on_reform_passed_effect = { amount = imperial_authority_uncontroversial_law }
			custom_tooltip = hre_golden_bull_formalizes_electorate_tt
			custom_tooltip = hre_golden_bull_self_voting_regulated_via_electorate_law
		}
		country_modifier = {
			#Emperor Modifier
			potential_trigger = {
				is_leader_of_international_organization = scope:recipient
			}
			monthly_imperial_authority = monthly_imperial_authority_mild_bonus
		}
		international_organization_modifier = {
			hre_enable_leave_hre_peace_treaty = yes
