#Agendas interacting with Institutions
pa_resist_feudalism = {
	estate = burghers_estate
	estate = peasants_estate

	potential = {
		knows_about_institution = institution:feudalism
		NOT = {
			has_embraced_institution = institution:feudalism
		}
		any_owned_location = {
			institution_progress:legalism > 0
			institution_progress:legalism < 99
		}
	}

	on_accept = {
		ordered_owned_location = {
			limit = {
				institution_progress:feudalism > 0
				institution_progress:feudalism < 99
			}
			order_by = population
			max = {
				value = 1
				add = {
					value = num_provinces
					divide = 20
				}
			}
			change_institution_progress = {
				type = institution:feudalism
				value = institution_progress_mild_penalty
			}
		}
	}
	chance = 10
}
