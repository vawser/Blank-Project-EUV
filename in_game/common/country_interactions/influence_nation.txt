influence_nation =  { 
	type = diplomacy
	category = CATEGORY_INFLUENCE_ACTIONS

	potential = {
		scope:actor = {
			modifier:allow_diplomacy_influence_nation = yes
		}
	}
	
	allow = {
	}
	
	ai_tick = daily
	ai_tick_frequency = 180
	
	select_trigger = {
		looking_for_a = country
		target_flag = recipient
		name = "choose_country"
		column = {
			data = name
		}
		visible = {
			is_great_power = no
		}
        enabled = {
		}
	}

	effect = {
		if = {
			limit = {
				exists = scope:recipient
			}
			scope:recipient = {
				add_opinion = {
					target = scope:actor
					modifier = opinion_opinion_influenced
				}
				add_trust = {
					target = scope:actor
					modifier = trust_opinion_influenced
				}
				
				add_favors = {
					target = scope:actor
					value = 5
				}
				add_location_modifier = {
					modifier = excellent_diplomacy
					years = 10
					mode = replace
				}
			}
		}
	}

	ai_will_do = {
		if = {
			limit = {
				scope:actor = {
					wants_opinion_with = scope:recipient
				}
			}
			add = 10
		}
	}
}
