	category = CATEGORY_SUBJECT_ACTIONS

	potential = {
	}

	allow = {
	}

		interaction_source_list = {
			scope:actor = {

					add_to_list = source
				}
			}
		}
		target_flag = recipient
		name = "choose_country"
		none_available_msg_key = "no_countries_available"
		column = {
			data = name
		}
		visible = {
			any_possible_policy = {

				}
			}
		}

		none_available_msg_key = "no_valid_policies"
		column = {
			data = name
		}
		visible = {
			scope:recipient = {
				any_possible_policy = {
					this = root
				}
			}
		}
		enabled = {
		}
	}

	effect = {
		scope:recipient = {	
			add_policy = scope:target
			add_country_modifier = {
				modifier = forced_us_to_use_their_policy
				years = 50
				mode = add_and_extend
			}
		}
		
	}
