junior_partner = {
	base = union_partner
	conditions = {
		PREV = { save_temporary_scope_as = senior_partner }
		union ?= {
			country_has_special_status = {
				type = special_status:senior_partner
				country = scope:senior_partner
			}
			NOT = {
				country_has_special_status = {
					type = special_status:senior_partner
					country = PREV
				}
			}
		}
	}
}

american_colonial_country = {
	base = country
	conditions = {
		NOT = { culture = { is_culture_native_american =yes } }
		capital.continent = continent:america
		is_subject_type = colonial_nation
	}
}

cardinal_country = {
	base = country_in_religion
	conditions = {
		num_cardinals > 0
	}
}
