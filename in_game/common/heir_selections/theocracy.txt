bishopric_elective = {
	include_ruler_siblings = no
	all_in_country = yes
	through_female = no
	allow_female = no
	ignore_ruler = yes
	cached = yes
	allow_foreign_ruler = no
	custom_tags = { empty_heir_religion_law }
	allowed_estates = { clergy_estate }
	heir_is_allowed = {
		scope:target = {
			has_variable_list = supported_heirs
			is_target_in_variable_list = {
				name = supported_heirs
				target = root
			}
		}
	}
	calc = {
		if = {
			limit = {
				exists = root
			}
			add = {
				desc = "HEIR_SELECTION_OLDER_HEIR_DESC"
				value = character_age
				multiply = 3
			}
			add = {
				desc = "HEIR_SELECTION_TOTAL_ABILITIES"
				value = total_abilities
			}
			if = { 
				limit = { modifier:blocked_from_being_ruler = yes }
				add = {
					desc = "HEIR_SELECTION_IS_UNSUITED_TO_RULE_DESC"
					value = -1000 
				}
			}
		}
		# Here it goes the description for explaining a heir gets +1 points per year.
		if = {
			limit = {
				not = { exists = root }
			}
			add = {
				desc = "HEIR_SELECTION_OLDER_HEIR_DESC"
				value = 1
			}
			add = {
				desc = "HEIR_SELECTION_TOTAL_ABILITIES"
				value = 1
			}
		}
	}
}

	}

		has_reform = government_reform:military_order_reform
	}

	locked = {
		has_reform = government_reform:military_order_reform
	}

	calc = {
		if = {

			add = {
				desc = "HEIR_SELECTION_OLDER_HEIR_DESC"

			if = {

				add = {
					desc = "HEIR_SELECTION_IS_UNSUITED_TO_RULE_DESC"
					value = -1000 
				}
			}
		}

		if = {
			limit = {
				exists = root
			}
			add = {

			if = {
				limit = {
					is_general = yes
				}

				value = 1
			}
			add = {
				desc = "HEIR_SELECTION_GENERAL_DESC"
				value = 1
			}
		}
	}
}
