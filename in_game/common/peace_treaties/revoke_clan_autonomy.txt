peace_revoke_clan_autonomy = {
	
	cost = {
		add = {
			desc = "DIPLOREASON_BASE"
			value = 50
		}

		add = {
			desc = "game_concept_population"
			value = scope:loser.total_population
			multiply = 0.009
		}
	}

	potential = {
		scope:loser = {
			country_type = location 
			is_situation_active = situation:sengoku
			num_locations < 5
		}
		scope:winner = { is_leader_of_international_organization = international_organization:japanese_shogunate }
		scope:war = {
			OR = {
				is_no_cb_war = yes
				casus_belli ?= casus_belli:cb_shogunate_reassert_authority
			}
		}
	}
	
	allow = {

	}

	effect = {
		scope:loser = {
			turn_japanese_clan_into_building_based = yes
		}
	}

	ai_desire = {
		if = {
			limit = {
				scope:winner = {
					is_rival_of = scope:loser
				}
			}
			add = {
				desc = "game_concept_rival"
				value = 10
			}
		}
		if = {
			limit = {
				scope:winner.prestige < scope:loser.prestige
			}
			add = {
				desc = "DIPLOREASON_LOWER_PRESTIGE"
				value = scope:loser.prestige
				subtract = scope:winner.prestige
			}
		}
	}
}
