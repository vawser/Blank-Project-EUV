league_of_public_weal_appanage_victory = {
	
	cost = {
		add = {
			desc = "DIPLOREASON_BASE"
			value = 50
		}
	}

	potential = {
		scope:war = {
			or = {
				has_casus_belli = no
				casus_belli = {
					is_trade_cb = no
				}
			}
		}
		has_global_variable = league_of_public_weal_enabled
		scope:loser = {
			tag = FRA
			can_make_subject_of = {
				target = scope:winner
				type = subject_type:appanage
			}
		}	
		scope:winner = {
			has_reform = government_reform:member_of_league_of_public_weal
		}	
	}
	
	allow = {
		scope:loser = { tag = FRA }
		scope:winner = { has_reform = government_reform:member_of_league_of_public_weal }
	}

	effect = {
		scope:winner = {
			custom_tooltip = weal_members_bonus_and_appanage_autonomy
			set_global_variable = league_of_public_weal_appanage_victory
			every_country = {
				limit = {
					has_reform = government_reform:member_of_league_of_public_weal
				}
				make_subject_of = {
					target = c:FRA
					type = subject_type:appanage
					war = scope:war
				}	
				add_country_modifier = { modifier = appanage_league_of_weal_victory years = 15 mode = add }
			}			
		}
		scope:loser = { 
			add_prestige = prestige_extreme_penalty 
		}
	}

	ai_desire = {
		value = 200
	}
}
