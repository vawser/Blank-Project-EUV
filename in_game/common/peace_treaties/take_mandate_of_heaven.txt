peace_mandate_of_heaven = {
	
	cost = {
		add = {
			desc = "DIPLOREASON_BASE"
			value = 5
		}
		add = {
			desc = "game_concept_population"
			value = scope:loser.total_population
			divide = scope:winner.total_population
			multiply = 5
		}
	}

	potential = {
		scope:war = {
			casus_belli ?= casus_belli:cb_claim_mandate_of_heaven
		}
		international_organization:middle_kingdom ?= { leader_country = scope:loser }
	}
	
	allow = {
	}

	effect = {
		international_organization:middle_kingdom = {
			if = {
				limit = {
					scope:winner = {
						NOT = { is_member_of_international_organization = prev }
					}
				}
				add_country_to_international_organization = scope:winner
			}
			set_leader_country = scope:winner
			if = {
				limit = {
					scope:loser = {
						NOT = { tag = CHI }
					}
				}
				scope:winner = {
					change_country_tag = CHI
				}
			}
		}
	}

	ai_desire = {
		if = {
			limit = {
				scope:winner = {
					is_rival_of = scope:loser
				}
			}
			add = {
				desc = "game_concept_rival"
				value = 10
			}
		}
		if = {
			limit = {
				scope:winner.prestige < scope:loser.prestige
			}
			add = {
				desc = "DIPLOREASON_LOWER_PRESTIGE"
				value = scope:loser.prestige
				subtract = scope:winner.prestige
			}
		}
	}
}
