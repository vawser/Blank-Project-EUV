claim_french_throne = {
	cost = {
		add = {
			desc = "DIPLOREASON_BASE"
			value = 70
		}
		add = {
			desc = strength_of_england_versus_france
			value = 100
			multiply = "scope:loser.relative_military_strength(scope:winner)"
		}
	}

	base_antagonism = 50
	antagonism_type = antagonism_assume_union_senority
	
	potential = {
		scope:war = {
			is_no_cb_war = no
			casus_belli ?= {
				OR = {
					this = casus_belli:cb_hundred_years_war
					this = casus_belli:cb_hundred_years_war_fra
				}
			}
		}
		scope:winner ?= {
			this = c:ENG
			NOT = { in_union_with = c:FRA }
		}
		scope:loser ?= c:FRA
	}
	allow = {
		region:france_region = {
			any_ownable_location_in_region = {
				is_owned_or_owned_by_subjects_or_below_of = scope:winner
				percent >= 0.15
			}
		}

		scope:loser = {
			capital ?= { controller = scope:winner }
		}

		scope:winner = {
			is_subject = no
			great_power_score > scope:loser.great_power_score
		}
	}

	effect = {
		if = {
			limit = {
				c:ENG ?= {
					union ?= {
						country_has_special_status = {
							type = special_status:senior_partner
							country = c:ENG
						}
					}
				}
			}
			c:ENG = { create_union = c:FRA }
		}
		else = {
			c:ENG = {
				create_union = c:FRA
				set_senior_partner = yes

				hidden_effect = {
					union ?= {
						add_policy_to_international_organization = policy:union_establish_seniority_policy
						international_organization_add_special_status = {
							type = special_status:senior_partner
							country = c:ENG
						}
					}
				}
			}
		}

		c:FRA = { change_heir_selection = c:ENG.succession_law }
	}

	ai_desire = {
		value = 9999
	}
}
