peace_leave_the_hre = {
	cost = {
		add = {
			desc = "DIPLOREASON_BASE"
			value = 15
		}
	}

	potential = {
		exists = international_organization:hre
		scope:loser = {
			is_emperor = yes
			is_in_same_international_organization = {
				international_organization = international_organization:hre
				target = scope:winner
			}
		}
	}
	
	allow = {
		international_organization:hre = {
			modifier:hre_enable_leave_hre_peace_treaty = yes
		}
	}

	effect = {
		scope:winner = {
			add_prestige = {
				value = scope:loser.prestige
				multiply =  0.1
			}
			international_organization:hre = {
				remove_country_from_international_organization = scope:winner
				remove_location_from_international_organization = scope:winner.capital
			}
		}
		scope:loser = { 
			add_prestige = prestige_severe_penalty
		}
		
	}

	ai_desire = {
		subtract = 100	#We do not want the AI to lose the HRE
	}
}
