peace_dismantle_hre = {
	cost = {
		add = {
			desc = "DIPLOREASON_BASE"
			value = 50
		}
	}
	potential = {
		exists = international_organization:hre
		scope:loser = {
            is_emperor = yes
		}
        scope:winner = {
            NOT = { is_member_of_international_organization = international_organization:hre }
        }
	}
	allow = {
		international_organization:hre = {
            any_international_organization_owned_location = {
                owner ?= { NOT = { is_member_of_international_organization = international_organization:hre } }
                percent >= 0.5
            }
		}
	}

	effect = {
		scope:winner = {
            destroy_international_organization = {
                target = international_organization:hre
            }
		}
		scope:loser = { 
			add_prestige = prestige_ultimate_penalty
		}
	}

	ai_desire = {
        if = {
            limit = {
                current_age_or_later = { age = age_6_revolutions }
            }
            add = 200   #it is time to say goodbye!
        }
        else = {
            subtract = 100  #hold it!
        }
	}
}
