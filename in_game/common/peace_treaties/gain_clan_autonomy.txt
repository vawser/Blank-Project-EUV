peace_gain_clan_autonomy = {
	
	cost = {
		add = {
			desc = "DIPLOREASON_BASE"
			value = 50
		}
	}

	potential = {
		scope:winner = {
			country_type = building 
			is_situation_active = situation:sengoku
			OR = {
				has_reform = government_reform:japanese_clan
				has_reform = government_reform:japanese_imperial_family
			}
		}
		international_organization:japanese_shogunate = { leader_country ?= scope:loser }
		scope:war = {
			or = {
				has_casus_belli = no
				casus_belli = {
					is_trade_cb = no
				}
			}
		}
	}
	
	allow = {

	}

	effect = {
		scope:winner = {
			turn_japanese_clan_into_normal_tag = yes
		}
	}

	ai_desire = {
		if = {
			limit = {
				scope:winner = {
					is_rival_of = scope:loser
				}
			}
			add = {
				desc = "game_concept_rival"
				value = 10
			}
		}
		if = {
			limit = {
				scope:winner.prestige < scope:loser.prestige
			}
			add = {
				desc = "DIPLOREASON_LOWER_PRESTIGE"
				value = scope:loser.prestige
				subtract = scope:winner.prestige
			}
		}
	}
}
