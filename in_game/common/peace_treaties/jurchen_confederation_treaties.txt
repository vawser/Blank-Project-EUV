	cost = 100

	potential = {
		scope:war = {

				has_casus_belli = no
				casus_belli = {
					is_trade_cb = no
				}
			}
		}
		scope:winner = {
			any_international_organizations_member_of = {
				international_organization_type = international_organization_type:jurchen_confederation
			}
		}
		scope:loser = {
			any_international_organizations_member_of = {
				international_organization_type = international_organization_type:jurchen_confederation
			}
		}
	}

	allow = {
		scope:winner = {
			any_international_organizations_member_of = {
				international_organization_type = international_organization_type:jurchen_confederation 
				leader_country ?= scope:winner
			}
		}
		scope:loser = {
			NOT = {
				any_international_organizations_member_of = {
					international_organization_type = international_organization_type:jurchen_confederation 
					leader_country ?= scope:loser
				}
			}
			any_international_organizations_member_of = {
				international_organization_type = international_organization_type:jurchen_confederation 
				NOT = { any_international_organization_member = { this = scope:winner } }
			}
		}
	}

	base_antagonism = 3
	antagonism_type = antagonism_jurchen_confederation_aggression

	effect = {
		scope:loser = {
			every_international_organizations_member_of = {
				limit = {
					international_organization_type = international_organization_type:jurchen_confederation
				}
				remove_country_from_international_organization = scope:loser
			}
		}
		scope:winner = {
			random_international_organizations_member_of = {
				limit = {
					international_organization_type = international_organization_type:jurchen_confederation
					leader_country ?= scope:winner
				}
				add_country_to_international_organization = scope:loser
			}
		}
	}

	ai_desire = 100
}
