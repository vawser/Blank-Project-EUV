steal_pops = {
	potential = {
		scope:winner = {
            country_type = pop
		}
        scope:loser = {
            country_type = pop
        }
	}

    allow = {
	}

	select_trigger = {
		looking_for_a = province_definition
		source = recipient
		target_flag = target
        visible = {
            scope:recipient = {
                any_owned_nomad_pop = {
                    location.province_definition = root
                }
            }
        }
    }

    cost = {
		add = {
			desc = "DIPLOREASON_BASE"
			value = 5
		}
		add = {
			desc = "game_concept_development"
			value = scope:target.development
			multiply = 0.1
		}
	}
	
	base_antagonism = 1
	antagonism_type = antagonism_steal_pops

	effect = {
		#transfer ownership of all pops in the province
        scope:target = {
            every_location_in_province_definition = {
                every_pop = {
                    limit = {
                        owner = scope:loser
                    }
                    change_pop_owner = scope:winner
                }
            }
        }
	}

	ai_desire = {
		
	}
}
