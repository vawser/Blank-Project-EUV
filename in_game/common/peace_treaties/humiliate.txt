peace_humiliate = {
	
	blocks_full_annexation = yes

	cost = {
		add = {
			desc = "DIPLOREASON_BASE"
			value = 5
		}
		add = {
			desc = "game_concept_prestige"
			value = scope:loser.prestige
			multiply = 0.5
		}
	}

	effect = {
		scope:winner = {
			add_prestige = {
				value = scope:loser.prestige
				multiply =  0.2
				min = 5
			}
			if = {
				limit = {
					scope:loser = { is_rival_of = scope:winner }
				}
				set_variable = {
					name = recently_humiliated_rival_variable
					years = 10
				}
			}
		}
		scope:loser = { 
			add_prestige = prestige_extreme_penalty 
		}
	}

	ai_desire = {
		if = {
			limit = {
				scope:winner = {
					is_rival_of = scope:loser
				}
			}
			add = {
				desc = "game_concept_rival"
				value = 10
			}
		}
		if = {
			limit = {
				scope:winner.prestige < scope:loser.prestige
			}
			add = {
				desc = "DIPLOREASON_LOWER_PRESTIGE"
				value = scope:loser.prestige
				subtract = scope:winner.prestige
			}
		}
	}
}
