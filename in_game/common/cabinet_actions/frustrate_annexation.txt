frustrate_annexation = {
	ability = adm
	
	potential = {
		OR = {
			is_subject = yes
			is_being_annexed = yes
		}
	}
	
	allow = {
		is_being_annexed = yes
		OR = {
			is_subject = no
			is_disloyal_subject = no
		}
	}

	is_finished = {
		OR = {
			is_being_annexed = no
			is_subject = no
		}
	}
	
	country_modifier = {
		diplomatic_annexation_cost = 0.5
	}
}
