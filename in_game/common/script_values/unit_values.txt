special_unit_victory_experience = {
	value = 1
	if = {
		limit = { exists = scope:land_units_killed }
		multiply = scope:land_units_killed
		divide = 100
		max = 100
	}
	else = {
		save_temporary_value_as = {
			name = land_units_killed	#For the error log
			value = 15
		}
		multiply = scope:land_units_killed
	}
}
