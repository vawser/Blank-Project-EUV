drill_army = {
	duration = -1
	toggle = yes
	army_only = yes
	cancel_on_combat_end = yes
	cancel_on_move = yes
	animation_gfx_override = 1
	
	map = yes
	

	allow = {
		is_army = yes
		in_combat = no
		is_moving = no
		is_exiled = no
		in_siege = no
		has_commander = yes

	}
	modifier = {
		monthly_experience_gain = 2.5
		army_maintenance_cost = 0.5
	}
	
	
	start_effect = {
	}
	
	
	
	finished_when = {
		OR = {
			in_combat = yes
			is_moving = yes
			is_exiled = yes
			in_siege = yes
			has_commander = no
		}
	}
	
	ai_will_do = {
		if = {
			limit = { 
				scope:owner = {
					at_war = no 
				}
				has_levies = no
				has_mercenaries = no
			}
			add = {
				value = 1
			}
		}
	}

	ai_will_revoke = {
		if = {
			limit = {
				or = {
					scope:owner = {
						at_war = yes 
					}
					has_levies = yes
					has_mercenaries = yes
				}
			}
			add = {
				value = 1
			}
		}
	}
}
