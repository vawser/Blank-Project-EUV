absorb_pops = {
	toggle = no
	army_only = yes
	
	map = yes
	
	hidden = {
		owner ?= {
			not = { country_type = pop }
		}
	}

	allow = {
		unit_location = { 
			any_pop = {
				culture = root.owner.culture
				religion = root.owner.religion
				owner ?= {
					not = { country_exists = this }
				}
			}
		}
	}
	
	start_effect = {
		unit_location = { 
			random_pop = {
				limit = {
					culture = root.owner.culture
					religion = root.owner.religion
					owner ?= {
						not = { country_exists = this }
					}
				}
				change_pop_owner = root.owner
			}
		}
	}
}
