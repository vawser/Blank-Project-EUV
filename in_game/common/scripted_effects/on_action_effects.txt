on_chinese_expedition_location_move = {
	if = {
		limit = {
			scope:expedition_initiator ?= {
				exists = var:chinese_treasure_fleet_location
				NOT = {
					has_discovered = scope:expedition_initiator.var:chinese_treasure_fleet_location
				}
			}
		}
		scope:expedition_initiator = {
			add_to_variable_list = {
				name = list_of_discovered_locations
				target = var:chinese_treasure_fleet_location
			}
		}
	}
}
