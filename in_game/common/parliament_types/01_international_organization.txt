### Early Imperial Diet (only Imperial princes and electors (not cities) can vote in the Imperial Diet)
# Historically the Imperial cities participation in the Imperial Diet and their voting power was unclear in the initial formation of the Imperial Diet, 
# Only with the "Reichsabschied von 1582" they got the "votum decisivum", to actually have their chamber count on the same level as the electoral chamber 
# Or the "Imperial Princely Chamber"
hre_early_imperial_diet = {
	type = international_organization
	potential = {
		international_organization_type = international_organization_type:hre
	}
	locked = { parliament_type = parliament_type:hre_early_imperial_diet }
	modifier = {
		has_international_parliament = yes
		uses_parliament_for_law_votes = yes
		forbid_multiple_policies_vote = yes
		elector_can_participate_in_parliament = yes
		archbishop_elector_can_participate_in_parliament = yes
		imperial_prince_can_participate_in_parliament = yes
	}
}
