muscovite_succession_war = {

	image = "gfx/interface/illustrations/disaster/time_of_troubles.dds"
	
	monthly_spawn_chance = monthly_spawn_chance_very_high

	can_start = {
		tag = MOS
		government_type = government_type:monarchy
		has_ruler = yes
		ruler = { has_dynasty = yes }
		any_rebel = {
			rebel_name_key = war_of_succession_pretender
			rebel_progress >= 25
	    }
		any_character = {
			is_alive = yes
			is_adult = yes
			is_female = no
			is_ruler = no
			dynasty = root.ruler.dynasty
		}
		
		legitimacy < 75

		has_any_active_disaster = no

		at_war = no
		
	}
	

	can_end = {
		muscovite_succession_war_end_trigger = yes
	}
	modifier = {
		monthly_legitimacy = -0.1
		stability_cost = 0.25
	}
	
	on_start = {
		trigger_event_non_silently = muscovite_succession_war.17
	}
	on_end = {
		trigger_event_non_silently = muscovite_succession_war.20
		clear_gui_variable = { type = muscovite_pretender_character }
	}
	
	on_monthly = {
		random_list = {
			5 = { custom_tooltip = an_event_occurs_tt }
			95 = {custom_tooltip = no_event_occurs_tt}
		}
		
		hidden_effect = {
			random_list = {
				3 = { trigger_event_silently =  muscovite_succession_war.18 }
				3 = { trigger_event_silently =  muscovite_succession_war.19 }
				94 = { }
			}
		}
	}
}
