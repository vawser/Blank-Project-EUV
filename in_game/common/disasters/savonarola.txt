savonarola_disaster = {

	image = "gfx/interface/illustrations/disaster/savonarola_disaster.dds"

	monthly_spawn_chance = monthly_spawn_chance_very_high

	can_start = {
		tag = FLO
		government_type = government_type:republic
		religion = religion:catholic
		current_year > 1470
		current_year < 1530		
		has_any_active_disaster = no
		OR = {
			estate_satisfaction:clergy_estate < 0.8
			societal_value:spiritualist_vs_humanist > 30
		}
		custom_tooltip = {
			text = had_csd_disaster_variable_tt
			NOT = { has_variable = had_savonarola_disaster }
		}
	}
	

	can_end = {
		savonarola_disaster_end_trigger = yes
	}
	modifier = {
		global_estate_target_satisfaction = big_permanent_target_satisfaction_penalty
		monthly_towards_traditionalist = societal_value_large_monthly_move
		stability_investment = large_stability_investment_penalty
	}
	
	on_start = {
		trigger_event_non_silently = savonarola.1
	}
	on_end = {
		set_variable = {
			name = had_savonarola_disaster
			years = 100
		}
		trigger_event_non_silently = savonarola.11
	}
	
	on_monthly = {
		random_list = {
			29 = { custom_tooltip = an_event_occurs_tt }
			71 = { custom_tooltip = no_event_occurs_tt }
		}
		hidden_effect = {
			random_list = {
		
				100 = { trigger_event_non_silently = savonarola.2 }
				100 = { trigger_event_non_silently = savonarola.3 }
				100 = { trigger_event_non_silently = savonarola.4 }
				100 = { trigger_event_non_silently = savonarola.5 }
				100 = { trigger_event_non_silently = savonarola.7 }
				100 = { trigger_event_non_silently = savonarola.8 }
				100 = { trigger_event_non_silently = savonarola.9 }
				100 = { trigger_event_non_silently = savonarola.10 }
				1000 = { custom_tooltip = no_event_occurs_tt }
			}
		}
	}
}
