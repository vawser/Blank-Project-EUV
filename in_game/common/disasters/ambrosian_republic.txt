ambrosian_republic_disaster = {

	image = "gfx/interface/illustrations/disaster/ambrosian_republic_disaster.dds"

	can_start = {
		tag = MLO
		government_type = government_type:monarchy

		OR = {
			has_regent = yes
			AND = {
				has_heir = no 
				ruler ?= { age_in_years >= 40 }
			}
		}
		legitimacy < 80

		has_any_active_disaster = no
		custom_tooltip = {
			text = had_csd_disaster_variable_tt
			NOR = { 
				has_variable = mlo_ambrosian_path 
				has_variable = chose_ally_union 
			}
		}
	}
	

	can_end = {
		ambrosian_republic_disaster_end_trigger = yes
	}
	modifier = {
		global_estate_target_satisfaction = large_permanent_target_satisfaction_penalty
		stability_cost = 0.25
	}
	
	on_start = {
		trigger_event_non_silently = flavor_mlo.1
	}
	on_end = {
		trigger_event_non_silently = ambrosian_republic.1
	}
	
	on_monthly = {
		
		random_list = {
			5 = {custom_tooltip = an_event_occurs_tt}
			95 = {custom_tooltip = no_event_occurs_tt}
		}
		hidden_effect = {
			random_list = {
				2 = { trigger_event_silently =  flavor_mlo.2 }
				2 = { trigger_event_silently =  flavor_mlo.31 }
				2 = { trigger_event_silently =  ambrosian_republic.5 }
				95 = { }
			}
		}
	}
}
