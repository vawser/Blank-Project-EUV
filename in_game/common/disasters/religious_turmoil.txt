religious_turmoil = {

	image = "gfx/interface/illustrations/disaster/religious_turmoil.dds"
	
	monthly_spawn_chance = monthly_spawn_chance_very_high



	can_start = {
		religion.group = religion_group:christian
		current_age = age_4_reformation
		is_subject = no
		NOT = { religion = religion:orthodox }
		NOT = { religion =  religion:miaphysite }
		
		reformation_is_enabled = yes

		situation:council_of_trent = {
			situation_has_ended = no
		}
		
		religious_unity < 0.80
		stability < 20
		societal_value:spiritualist_vs_humanist < 70
		
		NOT = { has_reform =government_reform:religious_tolerance }
		
		"estate_power(estate_type:clergy_estate)" < 0.20
		NOT = { tag = FRA }
	}
	

	can_end = {
		religious_turmoil_end_trigger = yes
	}
	modifier = {
		tolerance_heretic = -2
	}
	
	on_start = {
		trigger_event_non_silently = religious_turmoil.1
	}
	on_end = {
		trigger_event_non_silently = religious_turmoil.2
	}
	
	on_monthly = {

		show_as_tooltip = {
			96 = { custom_tooltip = no_event_occurs_tt }
			4 = { custom_tooltip = an_event_occurs_tt }
		}
		
		hidden_effect = {
			random_list = {
				960 = { }
				10 ={ trigger_event_silently = religious_turmoil.3 }
				10 ={ trigger_event_silently = religious_turmoil.4 }
				10 ={ trigger_event_silently = religious_turmoil.5 }
				10 ={ trigger_event_silently = religious_turmoil.6 }
			}
		}	
		
	}
}
