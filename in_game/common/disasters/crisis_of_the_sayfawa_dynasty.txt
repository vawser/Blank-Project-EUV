crisis_of_the_sayfawa_dynasty = {

	image = "gfx/interface/illustrations/disaster/decline_of_mali_disaster.dds"
	
	monthly_spawn_chance = monthly_spawn_chance_very_high

	can_start = {
		tag = KBO

		OR = {
			government_power < 90
			stability <= 30
		}
		has_any_active_disaster = no
		trigger_if = {
			limit = {
				OR = {
					has_ruler = yes
					has_regent = yes
				}
				ruler_or_regent = { has_dynasty = yes }
			}
			ruler_or_regent ?= { dynasty = dynasty:sayfawa_dynasty }
		}
		trigger_else = { always = no }
		custom_tooltip = { 
			text = had_csd_disaster_variable_tt 
			NOT = {
				has_variable = had_csd_disaster_variable 
			}
		}
	}
	

	can_end = {
		crisis_of_the_sayfawa_dynasty_end_trigger = yes
	}
	modifier = {
		global_estate_target_satisfaction = medium_permanent_target_satisfaction_penalty
		monthly_legitimacy = -0.3
		monthly_rebel_growth = 0.015
		global_crown_estate_power = -0.2
		
	}
	
	on_start = {
		trigger_event_non_silently = crisis_of_the_sayfawa_dynasty.1
	}
	on_end = {
		trigger_event_non_silently = crisis_of_the_sayfawa_dynasty.6
	}
	
	on_monthly = {

		random_list = {
			13 = { custom_tooltip = an_event_occurs_tt }
			87 = { custom_tooltip = no_event_occurs_tt }
		}

		hidden_effect = {
			random_list = {
		
				2500 = {}
				50 = { trigger_event_silently = crisis_of_the_sayfawa_dynasty.2 }
				90 = { trigger_event_silently = crisis_of_the_sayfawa_dynasty.3 }
				90 = { trigger_event_silently = crisis_of_the_sayfawa_dynasty.4 }
				90 = { trigger_event_silently = crisis_of_the_sayfawa_dynasty.5 }
				50 = { trigger_event_silently = crisis_of_the_sayfawa_dynasty.7 }
			}
			
		}
		
	}
}
