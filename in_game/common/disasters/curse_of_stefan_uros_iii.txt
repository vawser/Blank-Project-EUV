curse_of_stefan_uros_iii = {

	image = "gfx/interface/illustrations/disaster/curse_of_stefan_uros_iii.dds"
	
	monthly_spawn_chance = monthly_spawn_chance_very_high

	fire_only_once = yes

	can_start = {
		tag = SER
		custom_tooltip = {
			text = had_csd_disaster_variable_tt
			NOT = { has_variable = had_serbian_disaster_variable }
		}
		current_year < 1453
		country_rank = country_rank:rank_empire
		character:ser_stefan_dusan = {
			is_alive = no
		}
		ruler ?= { total_abilities <= 250 }
		stability <= 30
		government_power <= 75
	}
	

	can_end = {
		curse_of_stefan_uros_iii_end_trigger = yes
	}
	modifier = {
		monthly_pretender_rebel_growth  = 0.025
		monthly_nationalist_rebel_growth   = 0.025
		monthly_legitimacy = -0.25
	}
	
	on_start = {
		trigger_event_non_silently = curse_of_stefan_uros_iii.1
	}
	on_end = {
		trigger_event_non_silently = curse_of_stefan_uros_iii.2
	}
	
	on_monthly = {
		random_list = {
			20 = { custom_tooltip = an_event_occurs_tt }
			80 = { custom_tooltip = no_event_occurs_tt }
		}
		hidden_effect = {
			random_list = {
				1000 = {}
				50 = { trigger_event_silently = curse_of_stefan_uros_iii.3 }
				50 = { trigger_event_silently = curse_of_stefan_uros_iii.4 }
				50 = { trigger_event_silently = curse_of_stefan_uros_iii.5 }
				100 = { trigger_event_silently = curse_of_stefan_uros_iii.6 }
			}
		}
	}
}
