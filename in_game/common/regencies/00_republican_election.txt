republican_election_regency = {
	
	
	allow = {
		government_type = government_type:republic
	}
	
	modifier = {
		stability_investment = stability_investment_penalty
	}
	
	
	start_effect = {
		if = { 
			limit = { 
				any_character = {
					can_become_a_regent = yes
					has_estate = estate_type:burghers_estate
				}
			} 
			random_character = {
				limit = {
					can_become_a_regent = yes
					has_estate = estate_type:burghers_estate
				}
				save_scope_as = new_leader
			}
			set_regent = scope:new_leader
		}
		else = {
			hidden_effect = {
				create_character = {
					estate = estate_type:burghers_estate
					age = 35
					female = no
					save_scope_as = new_leader
				}
				set_regent = scope:new_leader
			}
		}

		
	}
	
}
