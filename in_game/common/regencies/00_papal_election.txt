papal_election_regency = {
	
	
	allow = {
		succession_law = heir_selection:papal_election
	}
	
	modifier = {
		stability_investment = stability_investment_penalty
	}
	
	
	start_effect = {
		if = { 
			limit = { 
				any_character = {
					can_become_a_regent = yes
					is_female = no
					has_estate = estate_type:clergy_estate
					religion = root.religion
				}
			} 
			random_character = {
				limit = {
					can_become_a_regent = yes
					is_female = no
					has_estate = estate_type:clergy_estate
					religion = root.religion
				}
				save_scope_as = new_leader
			}
			set_regent = scope:new_leader
		}
		else = {
			hidden_effect = {
				create_character = {
					religion = root.religion
					estate = estate_type:clergy_estate
					age = 35
					female = no
					save_scope_as = new_leader
				}
				set_regent = scope:new_leader
			}
		}
		religion:catholic = {
			add_religion_modifier = {
				modifier = papal_conclave_modifier
				days = -1
			}
		}

		
		
		trigger_event_silently = { id = papal_conclave_events.11  days = { 1 650 } }
	}
	
}
