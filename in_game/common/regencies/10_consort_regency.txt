consort_regency = {
	internally_assigned = yes
	
	allow = {
		has_consort = yes
		has_heir = yes
		consort ?= {
			is_adult = yes
			modifier:blocked_from_cabinet = no
		}
		NOT = { 
			religion.group = religion_group:muslim
			succession_law = heir_selection:tribal_matrilineal
		}
		NOR = {
			is_at_war_with = heir.employer
			is_at_war_with = consort.employer
		}
	}
	
	modifier = {
		counter_espionage = -0.05
		stability_investment = small_stability_investment
	}
	
	start_effect = {
		if = {
			limit = {
				exists = root.consort
			}
			set_regent = root.consort
		}
	}
	
}
