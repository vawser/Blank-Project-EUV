peasants_estate_regency = {
	
	
	allow = {
		"estate_power(estate_type:peasants_estate)" > 0.33
	}
	
	modifier = {
		peasants_estate_target_satisfaction = medium_permanent_target_satisfaction
		monthly_towards_free_subjects = societal_value_monthly_move
		global_nobles_estate_power = -0.25
		global_clergy_estate_power = -0.10
		global_burghers_estate_power = -0.10
	}
	
	start_effect = {
		if = { 
			limit = { 
				any_character = {
					can_become_a_regent = yes
					has_estate = estate_type:peasants_estate
				}
			} 
			random_character = {
				limit = {
					can_become_a_regent = yes
					has_estate = estate_type:peasants_estate
				}
				save_scope_as = new_leader
			}
			set_regent = scope:new_leader
		}
		else = {
			hidden_effect = {
				create_character = {
					estate = estate_type:peasants_estate
					age = 35
					female = no
					save_scope_as = new_leader
				}
			}
			set_regent = scope:new_leader
		}

		
	}
	
}
