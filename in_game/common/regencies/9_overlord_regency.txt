overlord_regency = {
	internally_assigned = yes
	
	allow = {
		is_subject = yes
		overlord = { 
			has_ruler = yes
			ruler ?= {
				can_become_a_regent = yes
				is_eligible_heir = root
			}
		}
	}
	
	modifier = {
		loyalty_to_overlord = 20
	}
	
	start_effect = {
		set_regent = root.overlord.ruler
		add_liberty_desire = liberty_desire_severe_minus
	}	
}
