interregnum = {
	
	
	allow = {
		always = yes
	}
	
	modifier = {
		stability_investment = stability_investment_penalty
	}
	
	
	start_effect = {
		if = { 
			limit = { 
				any_character = {
					can_become_a_regent = yes
					OR = {
						AND = {
							root.succession_law != heir_selection:tribal_matrilineal
							is_female = no
						}
						is_female = yes
					}
				}
			} 
			random_character = {
				limit = {
					can_become_a_regent = yes
					OR = {
						AND = {
							root.succession_law != heir_selection:tribal_matrilineal
							is_female = no
						}
						is_female = yes
					}
				}
				save_scope_as = new_leader
			}
		}
		else = {
			if = {
				limit = { root.succession_law = heir_selection:tribal_matrilineal }
				create_character = {
					age = 35
					female = yes
					religion = root.religion
					save_scope_as = new_leader
				}
			}
			else = {
				create_character = {
					age = 35
					female = no
					religion = root.religion
					save_scope_as = new_leader
				}
			}
		}

		set_regent = scope:new_leader
	}
	
}
