judicial_election_regency = {	
	allow = {
		succession_law = heir_selection:judicial_election
	}
	
	modifier = {
		stability_investment = stability_investment_penalty
	}
	
	
	start_effect = {
		if = { 
			limit = { 
				any_character = {
					can_become_a_regent = yes
					has_estate = estate_type:clergy_estate
					religion = root.religion
				}
			} 
			random_character = {
				limit = {
					can_become_a_regent = yes
					has_estate = estate_type:clergy_estate
					religion = root.religion
				}
				save_scope_as = new_leader
			}
			set_regent = scope:new_leader
		}
		else = {
			hidden_effect = {
				create_character = {
					religion = root.religion
					estate = estate_type:clergy_estate
					age = 35
					female = no
					save_scope_as = new_leader
				}
				set_regent = scope:new_leader
			}
		}

		
		
		trigger_event_non_silently = { id = kritarchy_events.1  days = { 1 600 } }
	}
	
}
