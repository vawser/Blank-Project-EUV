subject_regency = {
	internally_assigned = yes
	
	allow = {
		is_overlord = yes
		any_subject = {
			has_ruler = yes
			ruler ?= {
				can_become_a_regent = yes
				is_eligible_heir = root
			}

			great_power_score >= {
				value = root.great_power_score
				multiply = 0.5
			}

			country_rank_level >= root.country_rank_level
		}
	}
	
	modifier = {
		subject_loyalty = -20
		monthly_prestige = -0.1
	}
	
	start_effect = {
		random_subject = {
			limit = {
				has_ruler = yes
				ruler ?= {
					can_become_a_regent = yes
					is_eligible_heir = root
				}
				great_power_score >= {
					value = root.great_power_score
					multiply = 0.5
				}

				country_rank_level >= root.country_rank_level
			}
			add_liberty_desire = liberty_desire_severe_minus
			root = { set_regent = prev.ruler }
		}
	}	
}
