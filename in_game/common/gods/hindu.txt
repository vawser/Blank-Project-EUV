ganesha_god = {
	
	religion = hindu

	potential = {
		any_international_organizations_member_of = {
			international_organization_type = international_organization_type:hindu_branch
			international_organization_has_policy = policy:smartism
		}
	}

	country_modifier = {
		global_construction_speed = 0.05
		potential_trigger = {
			NOT = { has_access_to_god_holy_site = { god = ganesha_god } }
			smartism_balanced_gods = yes
			NOT = { is_favoring_god = { god = ganesha_god } }
		}
	}

	country_modifier = {
		potential_trigger = {
			OR = {
				is_favoring_god = { god = ganesha_god }
				AND = {
					has_access_to_god_holy_site = { god = ganesha_god }
					smartism_balanced_gods = yes
				}
			}
		}
		global_construction_speed = 0.1
		global_monthly_literacy = 0.01
	}

	country_modifier = {
		potential_trigger = {
			custom_tooltip = {
				text = disfavoured_god_tt
				smartism_balanced_gods = no
				NOT = { is_favoring_god = { god = ganesha_god } }
			}
		}
		global_construction_speed = -0.025
	}
}
