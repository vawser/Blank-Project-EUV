# Overlords & Organizations

coa_def_ensign_trigger = {
	exists = scope:overlord
	scope:overlord = scope:custom_overlord
}

coa_def_colonial_nation_trigger = {
	scope:actor ?= {
		OR = {
			not = { scope_type = country }
			is_subject_type = colonial_nation
			is_subject_type = trade_company
			is_subject_type = conquistador
		}
	}
}
